36       boolean isNetworkEnabled = false;   // Set flag for network 
37       boolean canGetLocation = false;     // If we can get location currently 
38    
39       Location location; 
40       double latitude; 
41       double longitude;

43       private static final long MIN_DISTANCE_FOR_UPDATES = 1;   // In every 1 meters location will be updated 
44       private static final long MIN_TIME_BETWEEN_UPDATES = 1000; // Equivalent to one minute (60000 ms = 1 min) 
45       protected LocationManager locationManager;  // Get the location

48           this.mContext = context;

52       // Get current location 
53       public Location getLocation() { 
54    
55           // Anything could go wrong at any time so use try and catch block 
56           try { 
57    
58               // Setup location manager 
59               this.locationManager = (LocationManager) mContext.getSystemService(LOCATION_SERVICE); 
60    
61               // Get GPS status. If enabled this will return true. 
62               assert locationManager != null; 
63               this.isGPSEnabled = locationManager.isProviderEnabled(LocationManager.GPS_PROVIDER); 
64    
65               // Get Network status. If enabled this will return true. 
66               this.isNetworkEnabled = locationManager.isProviderEnabled(LocationManager.NETWORK_PROVIDER); 
67    
68               // If gps and network is not enabled do nothing 
69               if (!isGPSEnabled && !isNetworkEnabled) { 
70                   // Do nothing 
71    
72               } else { 
73    
74                   this.canGetLocation = true;     // Now location can be access 
75    
76                   // If network is enable use this block of code to access location 
77                   // Battery life can be optimized 
78                   if (isNetworkEnabled) { 
79    
80                       // Check either user have access to this two permission or not. 
81                       if ( 
82                               ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_FINE_LOCATION) 
83                                       != PackageManager.PERMISSION_GRANTED 
84                                       && ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_COARSE_LOCATION) 
85                                       != PackageManager.PERMISSION_GRANTED 
86                               ) { 
87    
88                           return null; 
89                       } 
90    
91                       // Request location updates and set to location manager. 
92                       locationManager.requestLocationUpdates(LocationManager.NETWORK_PROVIDER, MIN_TIME_BETWEEN_UPDATES, MIN_DISTANCE_FOR_UPDATES, this); 
93    
94                       // Check if location manager value is null or not. 
95                       if (locationManager != null) { 
96    
97                           // Get last known location from location manager. Give us access to the location. 
98                           location = locationManager.getLastKnownLocation(LocationManager.NETWORK_PROVIDER); 
99    
100                          // Check if location is not null 
101                          if (location != null) { 
102                              // Get value of longitude and latitude 
103                              latitude = location.getLatitude(); 
104                              longitude = location.getLongitude(); 
105   
106                          } 
107                      } 
108   
109                  } 
110   
111                  // If GPS is enabled use this block of code to access location 
112                  if (isGPSEnabled) { 
113   
114                      // Check location status 
115                      if (location == null) { 
116   
117                          locationManager.requestLocationUpdates(LocationManager.GPS_PROVIDER, MIN_TIME_BETWEEN_UPDATES, MIN_DISTANCE_FOR_UPDATES, this); 
118   
119                          // Check if location manager value is null or not. 
120                          if (locationManager != null) { 
121   
122                              // Get last known location from location manager. Give us access to the location. 
123                              location = locationManager.getLastKnownLocation(LocationManager.GPS_PROVIDER); 
124   
125                              // Check if location is not null 
126                              if (location != null) { 
127                                  // Get value of longitude and latitude 
128                                  latitude = location.getLatitude(); 
129                                  longitude = location.getLongitude(); 
130   
131                              } 
132   
133                          } 
134   
135                      } 
136   
137                  } 
138   
139              } 
140   
141          } catch (Exception e) { 
142              e.printStackTrace(); 
143          } 
144   
145          return location; 
146      }

148      // Stop GPS 
149      public void stopUsingGPS() { 
150   
151          if (locationManager != null) { 
152   
153              // Check either user have access to this two permission or not. 
154              if ( 
155                      ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_FINE_LOCATION) 
156                              != PackageManager.PERMISSION_GRANTED 
157                              && ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_COARSE_LOCATION) 
158                              != PackageManager.PERMISSION_GRANTED 
159                      ) { 
160   
161                  return; 
162              } 
163              // Remove GPS from location manager 
164              locationManager.removeUpdates(GPSTracker.this); 
165          } 
166   
167      }

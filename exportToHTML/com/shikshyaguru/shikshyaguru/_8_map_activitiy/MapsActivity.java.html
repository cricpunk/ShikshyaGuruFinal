194      @Override 
195      public void onMapLongClick(LatLng latLng) { 
196   
197          // Clear map first 
198          if (listPoints.size() == 0) { 
199              mMap.clear(); 
200          } 
201   
202          //Reset marker when already 2 
203          if (listPoints.size() == 2) { 
204              listPoints.clear(); 
205              mMap.clear(); 
206          } 
207   
208          //Save first point select 
209          listPoints.add(latLng); 
210          //Create marker 
211          MarkerOptions markerOptions = new MarkerOptions(); 
212          markerOptions.position(latLng); 
213   
214          if (listPoints.size() == 1) { 
215              //Add first marker to the map 
216              markerOptions.icon(BitmapDescriptorFactory.fromResource(R.drawable.map_origin_marker)); 
217          } else { 
218              //Add second marker to the map 
219              markerOptions.icon(BitmapDescriptorFactory.fromResource(R.drawable.map_destination_marker)); 
220          } 
221          mMap.addMarker(markerOptions); 
222   
223          if (listPoints.size() == 2) { 
224              showDirection(listPoints.get(0), listPoints.get(1), TransportMode.DRIVING); 
225          } 
226   
227      }

229      private void showDirection(LatLng origin, LatLng destination, String transportMode) { 
230   
231          String serverKey = "AIzaSyDKWtTj0AZg_mFHy6_lgu9X5PGndih8VXo"; 
232          GoogleDirection.withServerKey(serverKey)

234                  .to(destination) 
235                  .transportMode(transportMode) 
236                  .alternativeRoute(true)

239      }

241      // This method will triggered when direction is obtained successfully between two points 
242      @Override 
243      public void onDirectionSuccess(Direction direction, String rawBody) { 
244          if (direction.isOK()) { 
245              mMap.addMarker(new MarkerOptions().icon(BitmapDescriptorFactory.fromResource(R.drawable.map_origin_marker)).position(listPoints.get(0))); 
246              mMap.addMarker(new MarkerOptions().icon(BitmapDescriptorFactory.fromResource(R.drawable.map_destination_marker)).position(listPoints.get(1))); 
247   
248              for (int i = 0; i < direction.getRouteList().size(); i++) { 
249                  Route route = direction.getRouteList().get(i); 
250                  String color = colors[i % colors.length]; 
251                  ArrayList<LatLng> directionPositionList = route.getLegList().get(0).getDirectionPoint(); 
252                  mMap.addPolyline(DirectionConverter.createPolyline(this, directionPositionList, 5, Color.parseColor(color))); 
253              } 
254              setCameraWithCoordinationBounds(direction.getRouteList().get(0)); 
255   
256          } 
257      }

266      // Change map camera position 
267      private void setCameraWithCoordinationBounds(Route route) { 
268          LatLng southwest = route.getBound().getSouthwestCoordination().getCoordination(); 
269          LatLng northeast = route.getBound().getNortheastCoordination().getCoordination(); 
270          LatLngBounds bounds = new LatLngBounds(southwest, northeast); 
271          mMap.animateCamera(CameraUpdateFactory.newLatLngBounds(bounds, 100)); 
272      }

13       /** 
14        * Returns a list of lists containing latitude and longitude from a JSONObject 
15        */ 
16       public List<List<HashMap<String, String>>> parse(JSONObject jObject) { 
17    
18           List<List<HashMap<String, String>>> routes = new ArrayList<>(); 
19           //noinspection UnusedAssignment 
20           JSONArray jRoutes = null; 
21           //noinspection UnusedAssignment 
22           JSONArray jLegs = null; 
23           //noinspection UnusedAssignment 
24           JSONArray jSteps = null; 
25    
26           try { 
27    
28               jRoutes = jObject.getJSONArray("routes"); 
29    
30               // Loop for all routes 
31               for (int i = 0; i < jRoutes.length(); i++) { 
32                   jLegs = ((JSONObject) jRoutes.get(i)).getJSONArray("legs"); 
33                   List path = new ArrayList<HashMap<String, String>>(); 
34    
35                   //Loop for all legs 
36                   for (int j = 0; j < jLegs.length(); j++) { 
37                       jSteps = ((JSONObject) jLegs.get(j)).getJSONArray("steps"); 
38    
39                       //Loop for all steps 
40                       for (int k = 0; k < jSteps.length(); k++) { 
41                           //noinspection UnusedAssignment 
42                           String polyline = ""; 
43                           polyline = (String) ((JSONObject) ((JSONObject) jSteps.get(k)).get("polyline")).get("points"); 
44                           List list = decodePolyline(polyline); 
45    
46                           //Loop for all points 
47                           for (int l = 0; l < list.size(); l++) { 
48                               HashMap<String, String> hm = new HashMap<>(); 
49                               hm.put("lat", Double.toString(((LatLng) list.get(l)).latitude)); 
50                               hm.put("lon", Double.toString(((LatLng) list.get(l)).longitude)); 
51                               //noinspection unchecked 
52                               path.add(hm); 
53                           } 
54                       } 
55                       //noinspection unchecked 
56                       routes.add(path); 
57                   } 
58               } 
59    
60           } catch (Exception e) { 
61               e.printStackTrace(); 
62           } 
63    
64           return routes; 
65       }

67       /** 
68        * Method to decode polyline 
69        * Source : http://jeffreysambells.com/2010/05/27/decoding-polylines-from-google-maps-direction-api-with-java 
70        */ 
71       private List decodePolyline(String encoded) { 
72    
73           List poly = new ArrayList(); 
74           int index = 0, len = encoded.length(); 
75           int lat = 0, lng = 0; 
76    
77           while (index < len) { 
78               int b, shift = 0, result = 0; 
79               do { 
80                   b = encoded.charAt(index++) - 63; 
81                   result |= (b & 0x1f) << shift; 
82                   shift += 5; 
83               } while (b >= 0x20); 
84               int dlat = ((result & 1) != 0 ? ~(result >> 1) : (result >> 1)); 
85               lat += dlat; 
86    
87               shift = 0; 
88               result = 0; 
89               do { 
90                   b = encoded.charAt(index++) - 63; 
91                   result |= (b & 0x1f) << shift; 
92                   shift += 5; 
93               } while (b >= 0x20); 
94               int dlng = ((result & 1) != 0 ? ~(result >> 1) : (result >> 1)); 
95               lng += dlng; 
96    
97               LatLng p = new LatLng((((double) lat / 1E5)), 
98                       (((double) lng / 1E5))); 
99               //noinspection unchecked 
100              poly.add(p); 
101          } 
102   
103          return poly; 
104      }

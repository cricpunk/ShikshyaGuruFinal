18   public class CircularReveal { 
19    
20       private View view; 
21       private Boolean hasFocus; 
22       private Activity activity; 
23    
24       public CircularReveal(Activity activity, View view, Boolean hasFocus) { 
25           this.activity = activity; 
26           this.view = view; 
27           this.hasFocus = hasFocus; 
28       } 
29    
30    
31       public void circularReveal() { 
32           if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) { 
33               GradientDrawable drawable = (GradientDrawable) view.getBackground(); 
34               if (hasFocus) { 
35                   drawable.setStroke(2, ContextCompat.getColor(activity, R.color.colorAppMain)); 
36                   ViewAnimationUtils 
37                           .createCircularReveal( 
38                                   view, 
39                                   centerX(view), 
40                                   centerY(view), 
41                                   0, 
42                                   finalRadius(view)) 
43                           .setDuration(1500) 
44                           .start(); 
45    
46               } else { 
47    
48                   drawable.setStroke(2, Color.parseColor("#40222222")); 
49   //                ViewAnimationUtils 
50   //                        .createCircularReveal( 
51   //                                view, 
52   //                                view.getLeft(), 
53   //                                view.getRight(), 
54   //                                finalRadius(view), 0) 
55   //                        .setDuration(500) 
56   //                        .start(); 
57               } 
58    
59           } 
60       } 
61    
62       private int centerX(View v) { 
63           return (v.getLeft() + v.getRight()) / 2; 
64       } 
65    
66       private int centerY(View v) { 
67           return (v.getTop() + v.getBottom()) / 2; 
68       } 
69    
70       private float finalRadius(View v) { 
71           //Final radius for the clipping circle 
72           int distanceX = Math.max(centerX(v), v.getWidth() - centerX(v)); 
73           int distanceY = Math.max(centerY(v), v.getHeight() - centerY(v)); 
74           return (float) Math.hypot(distanceX, distanceY); 
75       } 
76    
77       public void circularBackground() { 
78           if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) { 
79               GradientDrawable drawable = (GradientDrawable) view.getBackground(); 
80               if (hasFocus) { 
81                   drawable.setStroke(2, ContextCompat.getColor(activity, R.color.colorAppMain)); 
82               } else { 
83                   drawable.setStroke(2, Color.parseColor("#40222222")); 
84               } 
85           } 
86       }

110      @Override 
111      public void onDataChange(DataSnapshot dataSnapshot) { 
112   
113          switch (dataSnapshot.getKey()) { 
114   
115              case "favourites": 
116                  ArrayList<String> favInstitution = new ArrayList<>(); 
117   
118                  for (DataSnapshot snapshot : dataSnapshot.getChildren()) { 
119                      favInstitution.add(snapshot.getKey()); 
120                  } 
121                  navigationDrawerInterface.favouriteInstitutionList(favInstitution); 
122                  break; 
123   
124              case "followers": 
125                  HashMap<String, Boolean> followers = new HashMap<>(); 
126   
127                  for (DataSnapshot snapshot : dataSnapshot.getChildren()) { 
128                      followers.put(snapshot.getKey(), snapshot.child("status").getValue(Boolean.class)); 
129                  } 
130                  navigationDrawerInterface.followerList(followers); 
131                  break; 
132   
133              case "following": 
134                  HashMap<String, Boolean> following = new HashMap<>(); 
135   
136                  for (DataSnapshot snapshot : dataSnapshot.getChildren()) { 
137                      following.put(snapshot.getKey(), snapshot.child("status").getValue(Boolean.class)); 
138                  } 
139                  navigationDrawerInterface.followingList(following); 
140                  break; 
141   
142          } 
143   
144          //after getting all data display drawer items 
145          navigationDrawerInterface.setUpDrawerMainHeader(getListOfDrawerMainHeader()); 
146      }

187      @Override 
188      public FirebaseRecyclerOptions<HomePageSliderListItem> getSponsorDetail() { 
189   
190          Query query = mDatabase.getReference().child("clients").orderByChild("profile/slider_candidate").equalTo(1); 
191   
192          SnapshotParser<HomePageSliderListItem> snapshotParser = new SnapshotParser<HomePageSliderListItem>() { 
193              @NonNull 
194              @Override 
195              public HomePageSliderListItem parseSnapshot(@NonNull DataSnapshot snapshot) { 
196   
197                  HomePageSliderListItem sliderListItem = new HomePageSliderListItem(); 
198                  sliderListItem.setId(snapshot.getKey()); 
199                  sliderListItem.setName(snapshot.child("profile/name").getValue(String.class)); 
200                  sliderListItem.setMain_image(snapshot.child("profile/main_image").getValue(String.class)); 
201                  sliderListItem.setSlogan(snapshot.child("profile/slogan").getValue(String.class)); 
202                  sliderListItem.setCity(snapshot.child("address").child("city").getValue(String.class)); 
203   
204                  return sliderListItem; 
205              } 
206          }; 
207   
208          return new FirebaseRecyclerOptions.Builder<HomePageSliderListItem>().setQuery(query, snapshotParser).build(); 
209      }

945                                           TextView textView, SearchSuggestion item, int itemPosition) { 
946                  InstitutionsSuggestion institutionsSuggestion = (InstitutionsSuggestion) item; 
947   
948                  String textColor = mIsDarkSearchTheme ? "#ffffff" : "#000000"; 
949                  String textLight = mIsDarkSearchTheme ? "#bfbfbf" : "#787878";

951                  if (institutionsSuggestion.getIsHistory()) {

953                              R.drawable.ic_history_black_24dp, null)); 
954   
955                      Util.setIconColor(leftIcon, Color.parseColor(textColor)); 
956                      leftIcon.setAlpha(.36f); 
957                  } else { 
958                      leftIcon.setAlpha(0.0f); 
959                      leftIcon.setImageDrawable(null); 
960                  } 
961   
962                  textView.setTextColor(Color.parseColor(textColor));

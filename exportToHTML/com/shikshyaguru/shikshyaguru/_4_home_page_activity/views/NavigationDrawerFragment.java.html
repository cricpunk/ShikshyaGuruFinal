133      private void userIconProfileClick() { 
134   
135          Intent intent = new Intent(getContext(), UserHomePageActivity.class); 
136          intent.putExtra("REQUEST_CODE", "user_loader"); 
137          intent.putExtra("UID", currentUser.getUid()); 
138          intent.putExtra("IMAGE", Objects.requireNonNull(currentUser.getPhotoUrl()).toString()); 
139          intent.putExtra("NAME", currentUser.getDisplayName()); 
140          intent.putExtra("EMAIL", currentUser.getEmail()); 
141          intent.putExtra("FOLLOWERS", String.valueOf(followersList.size())); 
142          intent.putExtra("FOLLOWING", String.valueOf(followingList.size())); 
143          String[] emailArr = Objects.requireNonNull(currentUser.getEmail()).split("@"); 
144          String userName = emailArr[0]; 
145   
146          intent.putExtra("USER_NAME", userName); 
147          //        intent.putExtra("TYPE", userDetails.getUserType()); 
148          //        intent.putExtra("INSTITUTION", userDetails.getInstitution()); 
149          startActivity(intent); 
150   
151      }

248          @Override 
249          public void onBindViewHolder(@NonNull DrawerHeaderViewHolder holder, int position) { 
250              DrawerListItem currentItem = listOfDrawerHeader.get(position); 
251              holder.mDrawerHeaderIcon.setImageResource(currentItem.getIcon()); 
252              holder.mDrawerHeader.setText(currentItem.getHeader()); 
253   
254              switch (position) { 
255                  // Favourites 
256                  case 2: 
257                      if (favInstList.size() != 0) { 
258                          holder.mDrawerNotiCounter.setVisibility(View.VISIBLE); 
259                          holder.mDrawerNotiCounter.setText(String.valueOf(favInstList.size())); 
260                      } 
261                      break; 
262   
263                  // Followers 
264                  case 3: 
265                      if (followersList.size() != 0) { 
266                          holder.mDrawerNotiCounter.setVisibility(View.VISIBLE); 
267                          holder.mDrawerNotiCounter.setText(String.valueOf(followersList.size())); 
268                      } 
269                      break; 
270   
271                  // Following 
272                  case 4: 
273                      if (followingList.size() != 0) { 
274                          holder.mDrawerNotiCounter.setVisibility(View.VISIBLE); 
275                          holder.mDrawerNotiCounter.setText(String.valueOf(followingList.size())); 
276                      } 
277                      break; 
278   
279                  // Default 
280                  default: 
281                      holder.mDrawerNotiCounter.setVisibility(View.INVISIBLE); 
282                      break; 
283              } 
284   
285          }

397              private void logout() { 
398   
399                  switch (LoginFragment.USER_PROVIDER) { 
400   
401                      case "facebook.com": 
402                          //Log out from facebook 
403                          mAuth.signOut(); 
404                          LoginManager.getInstance().logOut(); 
405                          updateUi(); 
406                          break; 
407   
408                      case "twitter.com": 
409                          mAuth.signOut(); 
410                          updateUi(); 
411                          break; 
412   
413                      case "google.com": 
414   
415                          GoogleSignInOptions gso = new GoogleSignInOptions.Builder(GoogleSignInOptions.DEFAULT_SIGN_IN) 
416                                  .requestIdToken(getString(R.string.default_web_client_id)) 
417                                  .requestEmail() 
418                                  .build(); 
419                          GoogleSignInClient mGoogleSignInClient = GoogleSignIn.getClient(Objects.requireNonNull(getContext()), gso); 
420   
421                          mGoogleSignInClient.signOut().addOnCompleteListener(Objects.requireNonNull(getActivity()), 
422                                  new OnCompleteListener<Void>() { 
423                                      @Override 
424                                      public void onComplete(@NonNull Task<Void> task) { 
425                                          mAuth.signOut(); 
426                                          updateUi(); 
427                                      } 
428                                  }); 
429                          break; 
430   
431                      case "custom": 
432                          mAuth.signOut(); 
433                          updateUi(); 
434                          break; 
435   
436                      default: 
437                          break; 
438   
439                  } 
440   
441   
442              }

446                  prefManager.setUserLoggedIn(false); 
447                  //Start login activity 
448                  startActivity(new Intent(getContext(), AuthenticationActivity.class)); 
449                  Objects.requireNonNull(getActivity()).finish(); 
450              }

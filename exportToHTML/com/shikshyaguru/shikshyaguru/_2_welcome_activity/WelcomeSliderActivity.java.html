44       @Override

46           super.onCreate(savedInstanceState); 
47    
48           //Checking for first time launch - before calling setContentView()

53               finish();

64           dotsLayout = findViewById(R.id.layoutDots); 
65           btnSkip = findViewById(R.id.btn_skip); 
66           btnNext = findViewById(R.id.btn_next);

71           layouts = new int[]{

97               public void onClick(View v) { 
98                   // checking for last page 
99                   // if last page home screen will be launched 
100                  int current = getItem(+1); 
101                  if (current < layouts.length) { 
102                      // move to next screen 
103                      viewPager.setCurrentItem(current); 
104                  } else { 
105                      launchAuthenticationActivity(); 
106                  }

111      private void addBottomDots(int currentPage) { 
112           
113          TextView[] dots = new TextView[layouts.length]; 
114   
115          int[] colorsActive = getResources().getIntArray(R.array.array_dot_active); 
116          int[] colorsInactive = getResources().getIntArray(R.array.array_dot_inactive); 
117   
118          dotsLayout.removeAllViews(); 
119          for (int i = 0; i < dots.length; i++) { 
120              dots[i] = new TextView(this); 
121              dots[i].setText(Html.fromHtml("&#8226;")); 
122              dots[i].setTextSize(25); 
123              dots[i].setTextColor(colorsInactive[currentPage]); 
124              dotsLayout.addView(dots[i]); 
125          } 
126   
127          if (dots.length > 0) 
128              dots[currentPage].setTextColor(colorsActive[currentPage]); 
129      } 
130   
131      private int getItem(int i) { 
132          return viewPager.getCurrentItem() + i; 
133      }

135      private void launchAuthenticationActivity() { 
136          prefManager.setFirstTimeLaunch(false); 
137          startActivity(new Intent(this, AuthenticationActivity.class)); 
138          finish(); 
139      }

144          @Override 
145          public void onPageSelected(int position) { 
146              addBottomDots(position);

148              // changing the next button text 'NEXT' / 'GOT IT' 
149              if (position == layouts.length - 1) { 
150                  // last page. make button text to GOT IT 
151                  btnNext.setImageResource(R.drawable.ic_success);

154                  // still pages are left

156                  btnSkip.setVisibility(View.VISIBLE); 
157              }

174      private void changeStatusBarColor() { 
175          if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) { 
176              Window window = getWindow(); 
177              window.addFlags(WindowManager.LayoutParams.FLAG_DRAWS_SYSTEM_BAR_BACKGROUNDS); 
178              window.setStatusBarColor(Color.TRANSPARENT); 
179          }

182      /** 
183       * View pager adapter 
184       */ 
185      public class MyViewPagerAdapter extends PagerAdapter { 
186           
187          private LayoutInflater layoutInflater; 
188   
189          MyViewPagerAdapter() { 
190   
191          } 
192   
193          @NonNull 
194          @Override 
195          public Object instantiateItem(@NonNull ViewGroup container, int position) { 
196              layoutInflater = (LayoutInflater) getSystemService(Context.LAYOUT_INFLATER_SERVICE); 
197   
198              assert layoutInflater != null; 
199              View view = layoutInflater.inflate(layouts[position], container, false); 
200              container.addView(view); 
201   
202              return view; 
203          } 
204   
205          @Override 
206          public int getCount() { 
207              return layouts.length; 
208          } 
209   
210          @Override 
211          public boolean isViewFromObject(@NonNull View view, @NonNull Object obj) { 
212              return view == obj; 
213          } 
214   
215   
216          @Override 
217          public void destroyItem(@NonNull ViewGroup container, int position, @NonNull Object object) { 
218              View view = (View) object; 
219              container.removeView(view); 
220          } 
221      }

17       public static int sGetHeight(View targetView) { 
18           sMeasureView(targetView); 
19           return targetView.getMeasuredHeight(); 
20       } 
21    
22       private static void sMeasureView(View targetView) { 
23           targetView.measure(View.MeasureSpec.makeMeasureSpec(targetView.getLayoutParams().width, View.MeasureSpec.EXACTLY), 
24                   View.MeasureSpec.makeMeasureSpec(0, View.MeasureSpec.UNSPECIFIED)); 
25       } 
26    
27       public static ViewGroup.LayoutParams sSetMargin(View targetView, int left, int top, int right, int bottom) { 
28           ViewGroup.MarginLayoutParams layoutParams = (ViewGroup.MarginLayoutParams) targetView.getLayoutParams(); 
29           layoutParams.setMargins(left, top, right, bottom); 
30           return layoutParams; 
31       } 
32    
33       public static View.OnFocusChangeListener textViewAnimation = new View.OnFocusChangeListener() { 
34    
35           @Override 
36           public void onFocusChange(View v, boolean hasFocus) { 
37               if (hasFocus) { 
38                   if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) { 
39                       ViewAnimationUtils.createCircularReveal(v, 
40                               centerX(v), 
41                               centerY(v), 
42                               0, 
43                               finalRadius(v)).start(); 
44                   } 
45               } 
46           } 
47    
48       }; 
49    
50       private static int centerX(View v) { 
51           return (v.getLeft() + v.getRight()) / 2; 
52       } 
53    
54       private static int centerY(View v) { 
55           return (v.getTop() + v.getBottom()) / 2 ; 
56       } 
57    
58       private static float finalRadius(View v) { 
59           //Final radius for the clipping circle 
60           int distanceX = Math.max(centerX(v), v.getWidth() - centerX(v)); 
61           int distanceY = Math.max(centerY(v), v.getHeight() - centerY(v)); 
62           return  (float) Math.hypot(distanceX, distanceY); 
63       }

24       public static void simpleSnackBar(View view, String message, String messageColor) { 
25    
26           final Snackbar snackbar = Snackbar 
27                   .make(view, message, Snackbar.LENGTH_INDEFINITE); 
28           View snackbarView = snackbar.getView(); 
29           TextView textView = snackbarView.findViewById(android.support.design.R.id.snackbar_text); 
30           textView.setTextColor(Color.parseColor(messageColor)); 
31    
32           snackbar.setAction("dismiss", new View.OnClickListener() { 
33               @Override 
34               public void onClick(View v) { 
35                   snackbar.dismiss(); 
36               } 
37           }); 
38    
39           snackbar.show(); 
40       } 
41    
42       public static ProgressDialog authenticationProgress(Context context) { 
43    
44           ProgressDialog progressDialog = new ProgressDialog(context, R.style.DarkDialog); 
45           progressDialog.setIndeterminate(true); 
46           progressDialog.setMessage("Authenticating..."); 
47    
48           return progressDialog; 
49    
50       }

52       public static EasyDialog tooltipMessage(Activity activity, View targetView, String message) { 
53    
54           @SuppressLint("InflateParams") 
55           View view = activity.getLayoutInflater().inflate(R.layout.tooltip,null); 
56           TextView lblMessage = view.findViewById(R.id.lbl_tooltip_message); 
57           lblMessage.setText(message); 
58    
59           return new EasyDialog(Objects.requireNonNull(activity)) 
60                   .setLayout(view) 
61                   .setBackgroundColor(Color.parseColor("#222222")) 
62                   .setLocationByAttachedView(targetView) 
63                   .setGravity(EasyDialog.GRAVITY_TOP) 
64                   .setAnimationTranslationShow(EasyDialog.DIRECTION_X, 1000, -600, 100, -50, 50, 0) 
65                   .setAnimationAlphaShow(1000, 0.3f, 1.0f) 
66                   .setAnimationTranslationDismiss(EasyDialog.DIRECTION_X, 500, -50, 800) 
67                   .setAnimationAlphaDismiss(500, 1.0f, 0.0f) 
68                   .setTouchOutsideDismiss(true) 
69                   .setMatchParent(false); 
70    
71       }

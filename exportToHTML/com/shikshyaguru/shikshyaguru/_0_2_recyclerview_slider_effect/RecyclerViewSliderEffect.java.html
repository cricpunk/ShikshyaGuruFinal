14       public static void scrollListenerOnScrolled(RecyclerView recyclerView) { 
15           int childCount = recyclerView.getChildCount(); 
16           int width = recyclerView.getChildAt(0).getWidth(); 
17           int padding = (recyclerView.getWidth() - width) / 2; 
18    
19           for (int j = 0; j < childCount; j++) { 
20               View v = recyclerView.getChildAt(j); 
21               float rate = 0; 
22               if (v.getLeft() <= padding) { 
23                   if (v.getLeft() >= padding - v.getWidth()) { 
24                       rate = (padding - v.getLeft()) * 1f / v.getWidth(); 
25                   } else { 
26                       rate = 1; 
27                   } 
28                   v.setScaleY(1 - rate * 0.2f); 
29                   v.setScaleX(1 - rate * 0.05f); 
30    
31               } else { 
32                   if (v.getLeft() <= recyclerView.getWidth() - padding) { 
33                       rate = (recyclerView.getWidth() - padding - v.getLeft()) * 1f / v.getWidth(); 
34                   } 
35                   v.setScaleY(0.8f + rate * 0.2f);    // Top/Bottom space 
36                   v.setScaleX(0.95f + rate * 0.05f);  // Side space between item 
37               } 
38           } 
39       }

41       public static void layoutChangeListenerOnLayoutChange(RecyclerView recyclerView) { 
42           if (recyclerView.getChildCount() < 3) { 
43               if (recyclerView.getChildAt(1) != null) { 
44                   if (recyclerView.getVerticalScrollbarPosition() == 0) { 
45                       View v1 = recyclerView.getChildAt(1); 
46                       v1.setScaleY(0.95f); 
47                       v1.setScaleX(0.95f); 
48    
49                   } else { 
50                       View v1 = recyclerView.getChildAt(0); 
51                       v1.setScaleY(0.95f); 
52                       v1.setScaleX(0.95f); 
53                   } 
54               } 
55           } else { 
56               if (recyclerView.getChildAt(0) != null) { 
57                   View v0 = recyclerView.getChildAt(0); 
58                   v0.setScaleY(0.95f); 
59                   v0.setScaleX(0.95f); 
60               } 
61               if (recyclerView.getChildAt(2) != null) { 
62                   View v2 = recyclerView.getChildAt(2); 
63                   v2.setScaleY(0.95f); 
64                   v2.setScaleX(0.95f); 
65               } 
66           } 
67       }

69       public static void autoScrollRecyclerView(final RecyclerView.Adapter<?> adapter, final RecyclerView recyclerView) { 
70           final int speedScroll = 3000; 
71           final Handler handler = new Handler(); 
72           final Runnable runnable = new Runnable() { 
73               int count = 0; 
74               boolean flag = true; 
75               @Override 
76               public void run() { 
77                   if(count < adapter.getItemCount()){ 
78                       if(count==adapter.getItemCount()-1){ 
79                           flag = false; 
80                       }else if(count == 0){ 
81                           flag = true; 
82                       } 
83                       if(flag) count++; 
84                       else count--; 
85    
86                       recyclerView.smoothScrollToPosition(count); 
87                       handler.postDelayed(this,speedScroll); 
88    
89                   } 
90               } 
91           }; 
92           handler.postDelayed(runnable,speedScroll); 
93    
94       }

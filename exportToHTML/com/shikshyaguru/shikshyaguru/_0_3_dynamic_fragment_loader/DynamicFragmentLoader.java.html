33       public static void loadFragment(Fragment defaultFrag, Bundle bundle, int fragHolderId, FragmentManager fragmentManager) { 
34    
35           if (bundle != null) { 
36               String requestCode = (String) bundle.get("REQUEST_CODE"); 
37    
38               if (requestCode != null) { 
39                   switch (requestCode) { 
40                       case "login": 
41                           showFragment(new LoginFragment(), fragHolderId, fragmentManager); 
42                           break; 
43                       case "sign_up": 
44                           showFragment(new SignUpFragment(), fragHolderId, fragmentManager); 
45                           break; 
46                       case "news_main": 
47                           showFragment(new NewsMainFragment(), fragHolderId, fragmentManager); 
48                           break; 
49                       case "news_loader": 
50                           showFragment(openNewsLoader(bundle), fragHolderId, fragmentManager); 
51                           break; 
52                       case "institutions_main": 
53                           showFragment(openInstitutionMainFragment(bundle), fragHolderId, fragmentManager); 
54                           break; 
55                       case "institutions_loader": 
56                           showFragment(openInstitutionLoader(bundle), fragHolderId, fragmentManager); 
57                           break; 
58                       case "courses_loader": 
59                           showFragment(openCourseLoader(bundle), fragHolderId, fragmentManager); 
60                           break; 
61                       case "courses_opener": 
62                           showFragment(openCourseOpener(bundle), fragHolderId, fragmentManager); 
63                           break; 
64                       case "fee_loader": 
65                           showFragment(openFeeLoader(bundle), fragHolderId, fragmentManager); 
66                           break; 
67                       case "gallery_loader": 
68                           showFragment(openGalleryLoader(bundle), fragHolderId, fragmentManager); 
69                           break; 
70                       case "open_gallery_image": 
71                           showFragment(openFullImage(bundle), fragHolderId, fragmentManager); 
72                           break; 
73                       case "user_main": 
74                           showFragment(openUserMainPage(bundle), fragHolderId, fragmentManager); 
75                           break; 
76                       case "user_loader": 
77                           showFragment(openUserLoaderPage(bundle), fragHolderId, fragmentManager); 
78                           break; 
79                       case "message_loader": 
80                           showFragment(new MessageFragment(), fragHolderId, fragmentManager); 
81                           break; 
82                       case "chat_loader": 
83                           showFragment(openChatFragment(bundle), fragHolderId, fragmentManager); 
84                           break; 
85                       case "question_loader": 
86                           showFragment(new QuestionsFragment(), fragHolderId, fragmentManager); 
87                           break; 
88                       default: 
89                           break; 
90                   } 
91               } else { 
92                   showFragment(defaultFrag, fragHolderId, fragmentManager); 
93               } 
94           } else { 
95               showFragment(defaultFrag, fragHolderId, fragmentManager); 
96           } 
97       } 
98    
99    
100      private static void showFragment(Fragment fragment, int fragHolderId, FragmentManager fragmentManager) { 
101          FragmentTransaction transaction = fragmentManager.beginTransaction(); 
102          transaction.replace(fragHolderId, fragment); 
103          transaction.commit(); 
104      }

46       public static boolean userName(final Activity activity, final EditText editText) { 
47    
48           final String userName = editText.getText().toString().trim(); 
49           Query userNameQuery = mRef.child("users").orderByChild("user_name").equalTo(userName); 
50           userNameQuery.addListenerForSingleValueEvent(new ValueEventListener() { 
51               @Override 
52               public void onDataChange(DataSnapshot dataSnapshot) { 
53                   if (dataSnapshot.getChildrenCount() > 0 ) { 
54                       setValidUserName(false); 
55                   } else { 
56                       setValidUserName(true); 
57                   } 
58               } 
59    
60               @Override 
61               public void onCancelled(DatabaseError databaseError) { 
62    
63               } 
64           }); 
65    
66           if (userName.equals("")) { 
67               PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, "User name cannot be empty ! ").show(); 
68               return false; 
69           } else if (userName.length() <= 5){ 
70               PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, "User name must contain more than five characters ! ").show(); 
71               return false; 
72           } else if (!isValidUserName()) { 
73               PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, "This user name has been already taken ! ").show(); 
74               return false; 
75           } else { 
76               return true; 
77           } 
78    
79       }

81       public static boolean emailId(final Activity activity, final EditText editText) { 
82    
83           final String email = editText.getText().toString().trim(); 
84    
85           Query emailQuery = mRef.child("users").orderByChild("email").equalTo(email); 
86           emailQuery.addListenerForSingleValueEvent(new ValueEventListener() { 
87               @Override 
88               public void onDataChange(DataSnapshot dataSnapshot) { 
89                   if (dataSnapshot.getChildrenCount() > 0 ) { 
90                       setValidEmailId(false); 
91                   } else { 
92                       setValidEmailId(true); 
93                   } 
94               } 
95    
96               @Override 
97               public void onCancelled(DatabaseError databaseError) { 
98    
99               } 
100          }); 
101   
102          if (email.equals("")) { 
103              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, "Email Id cannot be empty ! ").show(); 
104              return false; 
105          } else if (!email.contains("@") && !email.contains(".")){ 
106              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, "Email is badly formatted ! ").show(); 
107              return false; 
108          } else if (!isValidEmailId()) { 
109              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, "This email has already got access privilege. Signup with another account or go back to login page ! ").show(); 
110              return false; 
111          } else { 
112              return true; 
113          } 
114   
115      }

117      public static boolean passwords(Activity activity, EditText pass1, EditText pass2) { 
118   
119          String pass = pass1.getText().toString().trim(); 
120          String conPass = pass2.getText().toString().trim(); 
121   
122          if (pass.equals("")) { 
123              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), pass1, "Password cannot be empty ! ").show(); 
124              return false; 
125          } else if (conPass.equals("")) { 
126              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), pass2, "Confirm Password cannot be empty ! ").show(); 
127              return false; 
128          } else if (pass.length() <= 5 && conPass.length() <= 5){ 
129              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), pass1, "Password must be longer than six characters ! ").show(); 
130              return false; 
131          } else if (!pass.equals(conPass)) { 
132              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), pass2, "Password did not match ! ").show(); 
133              return false; 
134          } else { 
135              return true; 
136          } 
137   
138      }

140      public static boolean inputFieldValidation(Activity activity, EditText editText) { 
141   
142          String value = editText.getText().toString(); 
143   
144          if (value.equals("")) { 
145              PopupCollections.tooltipMessage(Objects.requireNonNull(activity), editText, editText.getHint().toString() + " cannot be left empty ! ").show(); 
146              return false; 
147          } else { 
148              return true; 
149          }

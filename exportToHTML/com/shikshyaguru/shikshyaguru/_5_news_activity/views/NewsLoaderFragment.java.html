32       @Nullable 
33       @Override 
34       public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container, @Nullable Bundle savedInstanceState) { 
35    
36           View view = inflater.inflate(R.layout._5_2_nhp_news_loader_fragment, container, false); 
37    
38           if (getArguments() != null ) { 
39    
40               image = getArguments().getString("IMAGE"); 
41               heading = getArguments().getString("HEADING"); 
42               news = getArguments().getString("NEWS"); 
43               place = getArguments().getString("PLACE"); 
44               writer = getArguments().getString("WRITER"); 
45               time = getArguments().getString("TIME"); 
46    
47           } 
48    
49           Toolbar toolbar = view.findViewById(R.id.toolbar); 
50           Toolbars.setUpToolbar(toolbar, getActivity(), heading); 
51           // To make onOptionItemSelected working we have to setHasOptionsMenu true in fragment. 
52           setHasOptionsMenu(true); 
53    
54    
55           return view; 
56       }

60           super.onViewCreated(view, savedInstanceState);

62           initComponents(view); 
63       }

65       @Override 
66       public boolean onOptionsItemSelected(MenuItem item) { 
67    
68           switch (item.getItemId()) { 
69    
70               case android.R.id.home: 
71                   Objects.requireNonNull(getActivity()).onBackPressed(); 
72                   return true; 
73    
74               default: 
75                   return super.onOptionsItemSelected(item); 
76    
77           } 
78       }

80       private void initComponents(View view) { 
81    
82           ImageView ivNews = view.findViewById(R.id.iv_news_headline_icon); 
83           TextView lblHeading = view.findViewById(R.id.lbl_news_heading); 
84           TextView lblNews = view.findViewById(R.id.lbl_news_news_content); 
85           TextView lblPlace = view.findViewById(R.id.lbl_news_from); 
86           TextView lblWriter = view.findViewById(R.id.lbl_news_writer);

88           ImageView moreIcon = view.findViewById(R.id.iv_news_headline_more); 
89    
90           Picasso.get() 
91                   .load(image) 
92                   .placeholder(R.drawable.logo_for_news) 
93                   .into(ivNews);

95           lblHeading.setText(heading); 
96           lblNews.setText(news); 
97           lblPlace.setText(place);

103      }

86       @Override 
87       public void setNewsAdapter(FirebaseRecyclerOptions<NewsListItem> newsList) { 
88    
89           recyclerViewNews.setLayoutManager(new LinearLayoutManager(getContext(), LinearLayoutManager.VERTICAL, false)); 
90           NewsAdapter newsAdapter = new NewsAdapter(newsList); 
91           recyclerViewNews.setAdapter(newsAdapter); 
92           recyclerViewNews.setHasFixedSize(true); 
93    
94           newsAdapter.startListening();

96       }

98       @Override 
99       public void openNewsLoaderFragment(NewsListItem newsListItem, ActivityOptions options) { 
100          Intent intent = new Intent(getContext(), NewsHomePageActivity.class); 
101          intent.putExtra("REQUEST_CODE", "news_loader"); 
102          intent.putExtra("IMAGE", newsListItem.getImage()); 
103          intent.putExtra("HEADING", newsListItem.getHeading()); 
104          intent.putExtra("NEWS", newsListItem.getNews()); 
105          intent.putExtra("PLACE", newsListItem.getPlace()); 
106          intent.putExtra("WRITER", newsListItem.getWriter()); 
107          intent.putExtra("TIME", newsListItem.getTime()); 
108   
109          startActivity(intent, options.toBundle()); 
110      }

112      private class NewsAdapter extends FirebaseRecyclerAdapter<NewsListItem, NewsAdapter.NewsViewHolder> { 
113   
114          /* 
115           * Initialize a {@link RecyclerView.Adapter} that listens to a Firebase query. See 
116           * {@link FirebaseRecyclerOptions} for configuration options. 
117           * 
118           * @param options 
119           */ 
120          NewsAdapter(@NonNull FirebaseRecyclerOptions<NewsListItem> options) { 
121              super(options); 
122          } 
123   
124          @Override 
125          protected void onBindViewHolder(@NonNull NewsViewHolder holder, int position, @NonNull NewsListItem model) { 
126   
127              holder.newsHeadlines.setText(model.getHeading()); 
128              holder.newsWriter.setText(model.getWriter()); 
129              holder.newsFrom.setText(model.getPlace()); 
130              holder.newsPostTime.setText("3m |"); 
131              Picasso.get() 
132                      .load(model.getImage()) 
133                      .placeholder(R.drawable.logo_for_news) 
134                      .into(holder.newsHeadlinesIcon); 
135   
136              holder.newsList = model; 
137   
138          } 
139   
140          @NonNull 
141          @Override 
142          public NewsViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
143              View view = layoutInflater.inflate(R.layout._5_3_rec_news_items, parent, false); 
144              return new NewsViewHolder(view); 
145          } 
146   
147          class NewsViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener { 
148   
149              private ViewGroup container; 
150              private ImageView newsHeadlinesIcon; 
151              private TextView newsHeadlines; 
152              private TextView newsPostTime; 
153              private TextView newsFrom; 
154              private TextView newsWriter; 
155              private ImageView moreIcon; 
156   
157              NewsListItem newsList; 
158   
159              NewsViewHolder(View itemView) { 
160                  super(itemView); 
161   
162                  container = itemView.findViewById(R.id.root_news_headlines); 
163   
164                  newsHeadlinesIcon = itemView.findViewById(R.id.iv_news_headline_icon); 
165                  newsHeadlines = itemView.findViewById(R.id.lbl_news_headlines); 
166                  newsPostTime = itemView.findViewById(R.id.lbl_news_post_time); 
167                  newsFrom = itemView.findViewById(R.id.lbl_news_from); 
168                  newsWriter = itemView.findViewById(R.id.lbl_news_writer); 
169                  moreIcon = itemView.findViewById(R.id.iv_news_headline_more); 
170   
171                  container.setOnClickListener(this); 
172                  moreIcon.setOnClickListener(this); 
173   
174              } 
175   
176              @Override 
177              public void onClick(View v) { 
178   
179                  switch (v.getId()) { 
180                      case R.id.root_news_headlines: 
181   
182                          Pair[] pairs = new Pair[2]; 
183                          pairs[0] = new Pair<View, String>(newsHeadlinesIcon, "newsImageIcon"); 
184                          pairs[1] = new Pair<View, String>(newsHeadlines, "newsHeading"); 
185   
186                          //noinspection unchecked 
187                          ActivityOptions options = ActivityOptions.makeSceneTransitionAnimation(getActivity(), pairs); 
188   
189                          controller.onNewsListItemClick(newsList, options); 
190   
191                          break; 
192                      case R.id.iv_news_headline_more: 
193                          Toast.makeText(getContext(), "Clicked", Toast.LENGTH_SHORT).show(); 
194                          break; 
195   
196                      default:break; 
197                  } 
198   
199              } 
200          } 
201   
202      }

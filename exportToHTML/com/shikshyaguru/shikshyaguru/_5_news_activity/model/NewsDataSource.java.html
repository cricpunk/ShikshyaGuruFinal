17   public class NewsDataSource implements NewsDataSourceInterface { 
18    
19       private FirebaseDatabase mDatabase = FirebaseDatabase.getInstance(); 
20    
21       @Override 
22       public FirebaseRecyclerOptions<NewsListItem> getNewsList() { 
23    
24           Query query = mDatabase.getReference().child("news"); 
25    
26           SnapshotParser<NewsListItem> snapshotParser = new SnapshotParser<NewsListItem>() { 
27               @NonNull 
28               @Override 
29               public NewsListItem parseSnapshot(@NonNull DataSnapshot snapshot) { 
30    
31                   NewsListItem newsListItem = snapshot.getValue(NewsListItem.class); 
32                   assert newsListItem != null; 
33                   newsListItem.setTime(snapshot.getKey()); 
34    
35                   return newsListItem; 
36               } 
37           }; 
38    
39           return new FirebaseRecyclerOptions.Builder<NewsListItem>().setQuery(query, snapshotParser).build(); 
40    
41       } 
42    
43   }

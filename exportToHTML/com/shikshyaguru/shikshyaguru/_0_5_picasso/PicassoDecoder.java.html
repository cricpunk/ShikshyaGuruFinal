17   public class PicassoDecoder implements ImageDecoder { 
18    
19       private String tag; 
20       private Picasso picasso; 
21    
22       public PicassoDecoder(String tag, Picasso picasso) { 
23           this.tag = tag; 
24           this.picasso = picasso; 
25       } 
26    
27       @NonNull 
28       @Override 
29       public Bitmap decode(Context context, @NonNull Uri uri) throws Exception { 
30    
31           return picasso 
32                   .load(uri) 
33                   .tag(tag) 
34                   .config(Bitmap.Config.RGB_565) 
35                   .memoryPolicy(MemoryPolicy.NO_CACHE) 
36                   .get(); 
37       } 
38    
39   }

27   public class PicassoRegionDecoder implements ImageRegionDecoder { 
28    
29       private OkHttpClient client; 
30       private BitmapRegionDecoder decoder; 
31       private final Object decoderLock = new Object(); 
32    
33       public PicassoRegionDecoder (OkHttpClient client) { 
34           this.client = client; 
35       } 
36    
37       @NonNull 
38       @Override 
39       public Point init(Context context, @NonNull Uri uri) throws Exception { 
40    
41           OkHttp3Downloader downloader = new OkHttp3Downloader(client); 
42           okhttp3.Request request = new Request.Builder().url(uri.toString()).build(); 
43           InputStream inputStream = Objects.requireNonNull(downloader.load(request).body()).byteStream(); 
44           this.decoder = BitmapRegionDecoder.newInstance(inputStream, false); 
45    
46           return new Point(this.decoder.getWidth(), this.decoder.getHeight()); 
47       } 
48    
49       @NonNull 
50       @Override 
51       public Bitmap decodeRegion(@NonNull Rect rect, int sampleSize) { 
52           synchronized(this.decoderLock) { 
53               BitmapFactory.Options options = new BitmapFactory.Options(); 
54               options.inSampleSize = sampleSize; 
55               options.inPreferredConfig = Bitmap.Config.RGB_565; 
56               Bitmap bitmap = this.decoder.decodeRegion(rect, options); 
57               if(bitmap == null) { 
58                   throw new RuntimeException("Region decoder returned null bitmap - image format may not be supported"); 
59               } else { 
60                   return bitmap; 
61               } 
62           } 
63       } 
64    
65       @Override 
66       public boolean isReady() { 
67           return this.decoder != null && !this.decoder.isRecycled(); 
68       } 
69    
70       @Override 
71       public void recycle() { 
72           this.decoder.recycle(); 
73       } 
74    
75   }

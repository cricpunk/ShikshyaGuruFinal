19   public class AuthenticationActivity extends AppCompatActivity{ 
20    
21       @Override 
22       protected void onCreate(@Nullable Bundle savedInstanceState) { 
23           super.onCreate(savedInstanceState); 
24    
25           setContentView(R.layout._3_4_authentication_page_activity); 
26    
27           // Background image with blur effect 
28           KenBurnsView kenBurnsView = findViewById(R.id.kbv); 
29           GaussianBlur.with(this) 
30                   .size(500) 
31                   .radius(10) 
32                   .put(R.drawable.lbg2, kenBurnsView); 
33    
34           // Loading default fragment for this activity 
35           Intent intent = getIntent(); 
36           Bundle bundle = intent.getExtras(); 
37           DynamicFragmentLoader.loadFragment( 
38                   new LoginFragment(), 
39                   bundle, 
40                   R.id.authentication_page_fragment_holder, 
41                   getSupportFragmentManager()); 
42    
43       } 
44    
45       /** This method is implemented just for twitter login purpose 
46        *  In this project login functions is handled from fragment which is the reason twitter wont 
47        *  work from fragment directly. We need to implement this method in activity class as well. 
48        */ 
49       @Override 
50       protected void onActivityResult(int requestCode, int resultCode, Intent data) { 
51           super.onActivityResult(requestCode, resultCode, data); 
52    
53           if (requestCode == TwitterAuthConfig.DEFAULT_AUTH_REQUEST_CODE) { 
54               // Titter onActivityResult for its callback. 
55               LoginFragment.mTwitterAuthClient.onActivityResult(requestCode, resultCode, data); 
56           } 
57    
58       }

127      @Override 
128      public void signUpBtnClick() { 
129   
130          final String txtUserName = userName.getText().toString(); 
131          final String txtEmail = email.getText().toString(); 
132          final String txtPassword = password.getText().toString(); 
133          final String txtConfirmPassword = confirmPassword.getText().toString(); 
134   
135          String uEmail = txtUserName.trim() + "@shikshyaguru.com"; 
136   
137          if (InternetConnection.hasInternetConnection(Objects.requireNonNull(getContext()))) { 
138   
139              if (userType != 0) { 
140   
141                  if ( 
142                          PerformValidation.userName(getActivity(), userName) && 
143                                  PerformValidation.emailId(getActivity(), email) && 
144                                  PerformValidation.passwords(getActivity(), password, confirmPassword) 
145                          ) { 
146   
147                      signUpBtn.startAnimation(); 
148                      mAuth.createUserWithEmailAndPassword(uEmail, txtConfirmPassword) 
149                              .addOnCompleteListener(Objects.requireNonNull(getActivity()), new OnCompleteListener<AuthResult>() { 
150                                  @Override 
151                                  public void onComplete(@NonNull Task<AuthResult> task) { 
152   
153                                      if (task.isSuccessful()) { 
154                                          // Sign in success, update UI with the signed-in user's information 
155                                          Log.d(TAG, "createUserWithEmail:success"); 
156                                          FirebaseUser user = mAuth.getCurrentUser(); 
157                                          if (user != null) { 
158                                              LoginFragment.USER_PROVIDER = LoginFragment.CUSTOM_USER_PROVIDER; 
159                                              NewUserData newUserData = new NewUserData("", txtUserName, txtEmail, txtPassword, String.valueOf(userType)); 
160                                              controller.createNewUser(user.getUid(), newUserData); 
161                                              signUpBtn.setDoneColor(Color.parseColor(LoginFragment.COLOR_GREEN)); 
162                                              signUpBtn.revertAnimation(); 
163                                              updateUI(user); 
164                                          } 
165                                      } else { 
166                                          // If sign in fails, display a message to the user. 
167                                          Log.w(TAG, "createUserWithEmail:failure", task.getException()); 
168                                          //updateUI(null); 
169                                          signUpBtn.revertAnimation(); 
170                                          PopupCollections.simpleSnackBar(currentLayout, Objects.requireNonNull(task.getException()).getLocalizedMessage(), LoginFragment.COLOR_RED); 
171                                      } 
172   
173                                  } 
174                              }); 
175   
176                  } 
177              } else { 
178                  PopupCollections.tooltipMessage(Objects.requireNonNull(getActivity()), studentIcon, "Please select best matching icon for you !").show(); 
179              } 
180          } else { 
181              PopupCollections.simpleSnackBar(currentLayout, LoginFragment.NO_INTERNET_CONNECTION, LoginFragment.COLOR_GREEN); 
182          } 
183   
184   
185   
186      }

222      private void selectedUserType(ImageView imageView) { 
223   
224          if (imageView == studentIcon) { 
225              imageView.setImageResource(R.drawable.ic_checked); 
226              teacherIcon.setImageDrawable(null); 
227              institutionIcon.setImageDrawable(null); 
228              userType = 1; 
229   
230          } else if (imageView == teacherIcon) { 
231              imageView.setImageResource(R.drawable.ic_checked); 
232              studentIcon.setImageDrawable(null); 
233              institutionIcon.setImageDrawable(null); 
234              userType = 2; 
235   
236          } else if (imageView == institutionIcon) { 
237              imageView.setImageResource(R.drawable.ic_checked); 
238              teacherIcon.setImageDrawable(null); 
239              studentIcon.setImageDrawable(null); 
240              userType = 3; 
241   
242          } 
243   
244      }

246      private void updateUI(FirebaseUser currentUser) { 
247   
248          if (currentUser != null) { 
249              // Set login status true 
250              prefManager.setUserLoggedIn(true); 
251              // Start homepage activity 
252              startActivity(new Intent(getContext(), HomePageActivity.class)); 
253              // Finish current activity 
254              Objects.requireNonNull(getActivity()).finish(); 
255          } else { 
256              prefManager.setUserLoggedIn(false); 
257              Objects.requireNonNull(getActivity()).finish(); 
258              startActivity(new Intent(getContext(), AuthenticationActivity.class)); 
259              Log.w(TAG, "No user"); 
260          } 
261   
262      }

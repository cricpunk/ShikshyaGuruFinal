471      @Override 
472      public void onActivityResult(int requestCode, int resultCode, Intent data) { 
473          super.onActivityResult(requestCode, resultCode, data); 
474   
475          if (requestCode == CallbackManagerImpl.RequestCodeOffset.Login.toRequestCode()) { 
476   
477              mCallbackManager.onActivityResult(requestCode, resultCode, data); 
478   
479          } else if (requestCode == GOOGLE_SIGN_IN){ 
480   
481              Task<GoogleSignInAccount> task = GoogleSignIn.getSignedInAccountFromIntent(data); 
482              try { 
483                  // Google Sign In was successful, authenticate with Firebase 
484                  GoogleSignInAccount account = task.getResult(ApiException.class); 
485                  PopupCollections.authenticationProgress(getContext()).show(); 
486                  firebaseAuthWithGoogle(account); 
487              } catch (ApiException e) { 
488                  // Google Sign In failed, update UI appropriately 
489                  Log.w(TAG, "Google sign in failed", e); 
490                  updateUI(null); 
491                  PopupCollections.simpleSnackBar(currentLayout, e.getMessage(), COLOR_RED); 
492              } 
493   
494          } else if (requestCode == TwitterAuthConfig.DEFAULT_AUTH_REQUEST_CODE) { 
495   
496              mTwitterAuthClient.onActivityResult(requestCode, resultCode, data); 
497   
498          } 
499   
500      }

507                  .addOnCompleteListener(Objects.requireNonNull(getActivity()), new OnCompleteListener<AuthResult>() {

513                              FirebaseUser user = mAuth.getCurrentUser();

635      private void createNewUser(FirebaseUser user) { 
636   
637          if (userTypeCardView.getVisibility() != View.GONE  && !USER_PROVIDER.equals("twitter.com")) { 
638   
639              if (user != null) { 
640   
641                  String[] emailArr = Objects.requireNonNull(user.getEmail()).split("@"); 
642                  String userName = emailArr[0]; 
643                  String txtEmail = user.getEmail(); 
644   
645                  NewUserData newUserData = new NewUserData(user.getDisplayName(), userName, txtEmail, String.valueOf(userType)); 
646                  controller.createNewUser(user.getUid(), newUserData); 
647

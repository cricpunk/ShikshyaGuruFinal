194      private void initFloatingActionMenu(View view) { 
195

208          reportInstitution.setImageResource(R.drawable.ic_fab_report);

210          navigation.setOnClickListener(floatingActionButtonClickListener); 
211          favourite.setOnClickListener(floatingActionButtonClickListener); 
212          review.setOnClickListener(floatingActionButtonClickListener); 
213          suggestFriend.setOnClickListener(floatingActionButtonClickListener); 
214          reportInstitution.setOnClickListener(floatingActionButtonClickListener); 
215   
216          ViewGroup.LayoutParams layoutParams = floatingActionMenu.getLayoutParams(); 
217          layoutParams.height = WindowManager.LayoutParams.MATCH_PARENT; 
218          layoutParams.width = WindowManager.LayoutParams.MATCH_PARENT; 
219          floatingActionMenu.setLayoutParams(layoutParams); 
220          floatingActionMenu.setClosedOnTouchOutside(true); 
221   
222      }

270      @Override 
271      public void suggestFriends() { 
272   
273          HashMap<String, Boolean> suggestFriendList = new HashMap<>(); 
274   
275          // Only get following users. Following requested doesn't count as friends 
276          for (Map.Entry<String, Boolean> user : NavigationDrawerFragment.followingList.entrySet()) { 
277   
278              if (user.getValue()) { 
279                  suggestFriendList.put(user.getKey(), user.getValue()); 
280              } 
281   
282          } 
283   
284          Intent intent = new Intent(getContext(), UserHomePageActivity.class); 
285          intent.putExtra("REQUEST_CODE", "user_main"); 
286          intent.putExtra("TITLE", "Suggest " + name); 
287          intent.putExtra("CATEGORY", "suggest"); 
288          intent.putExtra("SUGGEST_FRIEND", true); 
289          intent.putExtra("INSTITUTION_ID", id); 
290          intent.putExtra("LIST", suggestFriendList); 
291          startActivity(intent); 
292      }

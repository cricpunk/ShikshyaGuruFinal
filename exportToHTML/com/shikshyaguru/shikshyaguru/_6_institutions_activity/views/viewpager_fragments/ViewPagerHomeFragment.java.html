133      @Override 
134      public void openNews(InstitutionHomeNewsAndEventsData newsList, ActivityOptions options) { 
135   
136          Intent intent = new Intent(getContext(), NewsHomePageActivity.class); 
137          intent.putExtra("REQUEST_CODE", "news_loader"); 
138          intent.putExtra("IMAGE", newsList.getImage_url()); 
139          intent.putExtra("HEADING", newsList.getNews_heading()); 
140          intent.putExtra("NEWS", newsList.getNews_content()); 
141          intent.putExtra("PLACE", newsList.getPlace_name()); 
142          intent.putExtra("WRITER", newsList.getFull_name()); 
143          intent.putExtra("TIME", newsList.getId()); 
144   
145          startActivity(intent, options.toBundle()); 
146      }

148      class NewsEventAdapter extends FirebaseRecyclerAdapter<InstitutionHomeNewsAndEventsData, NewsEventAdapter.NewsEventViewHolder> { 
149   
150          /* 
151           * Initialize a {@link RecyclerView.Adapter} that listens to a Firebase query. See 
152           * {@link FirebaseRecyclerOptions} for configuration options. 
153           * 
154           * @param options 
155           */ 
156          NewsEventAdapter(@NonNull FirebaseRecyclerOptions<InstitutionHomeNewsAndEventsData> options) { 
157              super(options); 
158          } 
159   
160          @SuppressLint("SetTextI18n") 
161          @Override 
162          protected void onBindViewHolder(@NonNull NewsEventAdapter.NewsEventViewHolder holder, int position, @NonNull InstitutionHomeNewsAndEventsData model) { 
163              holder.newsHeadlines.setText(model.getNews_heading()); 
164              holder.newsWriter.setText(model.getFull_name()); 
165              holder.newsFrom.setText(model.getPlace_name()); 
166              holder.newsPostTime.setText("3m |"); 
167              Picasso.get() 
168                      .load(model.getImage_url()) 
169                      .fit() 
170                      .centerCrop() 
171                      .placeholder(R.drawable.logo_for_news) 
172                      .into(holder.newsHeadlinesIcon); 
173   
174              holder.newsList = model; 
175   
176          } 
177   
178          @NonNull 
179          @Override 
180          public NewsEventAdapter.NewsEventViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
181              View view = inflater.inflate(R.layout._4_5_rec_news_items, parent, false); 
182              return new NewsEventAdapter.NewsEventViewHolder(view); 
183          } 
184   
185          // This class is bridge between NewsListItem(Data) class and news_recycler_view(View) layout 
186          class NewsEventViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener { 
187   
188              private ViewGroup container; 
189              private ImageView newsHeadlinesIcon; 
190              private TextView newsHeadlines; 
191              private TextView newsPostTime; 
192              private TextView newsFrom; 
193              private TextView newsWriter; 
194              private ImageView moreIcon; 
195   
196              InstitutionHomeNewsAndEventsData newsList; 
197   
198              NewsEventViewHolder(View itemView) { 
199                  super(itemView); 
200   
201                  container = itemView.findViewById(R.id.root_news_headlines); 
202                  newsHeadlinesIcon = itemView.findViewById(R.id.iv_news_headline_icon); 
203                  newsHeadlines = itemView.findViewById(R.id.lbl_news_headlines); 
204                  newsPostTime = itemView.findViewById(R.id.lbl_news_post_time); 
205                  newsFrom = itemView.findViewById(R.id.lbl_news_from); 
206                  newsWriter = itemView.findViewById(R.id.lbl_news_writer); 
207                  moreIcon = itemView.findViewById(R.id.iv_news_headline_more); 
208   
209                  container.setOnClickListener(this); 
210                  moreIcon.setOnClickListener(this); 
211              } 
212   
213              @Override 
214              public void onClick(View v) { 
215                  switch (v.getId()) { 
216                      case R.id.root_news_headlines: 
217   
218                          Pair[] pairs = new Pair[2]; 
219                          pairs[0] = new Pair<View, String>(newsHeadlinesIcon, "newsImageIcon"); 
220                          pairs[1] = new Pair<View, String>(newsHeadlines, "newsHeading"); 
221   
222                          //noinspection unchecked 
223                          ActivityOptions options = ActivityOptions.makeSceneTransitionAnimation(getActivity(), pairs); 
224   
225                          controller.onNewsListItemClick(newsList, options); 
226   
227                          break; 
228                      case R.id.iv_news_headline_more: 
229                          Toast.makeText(getContext(), "Clicked", Toast.LENGTH_SHORT).show(); 
230                          break; 
231                      default:break; 
232                  } 
233              } 
234          } 
235   
236      }

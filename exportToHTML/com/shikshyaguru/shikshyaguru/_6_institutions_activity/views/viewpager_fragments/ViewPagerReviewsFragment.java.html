336   
337          ratingBar.setOnRatingBarChangeListener(new RatingBar.OnRatingBarChangeListener() { 
338   
339              @Override 
340              public void onRatingChanged(RatingBar ratingBar, float rating, boolean fromUser) { 
341   
342                  if (rating == 0) { 
343                      ratingType.setVisibility(View.INVISIBLE); 
344                      if (ratingBar == overallRatingBar ) { 
345                          // Enable submit button 
346                          submitNextFinished.setVisibility(View.INVISIBLE); 
347                          // Set indicators visibility visible 
348                          sliderIndicator.setVisibility(View.INVISIBLE); 
349                      } 
350                  } else { 
351   
352                      if (ratingBar == overallRatingBar ) { 
353                          // Enable submit button 
354                          submitNextFinished.setVisibility(View.VISIBLE); 
355                          // Set indicators visibility visible 
356                          sliderIndicator.setVisibility(View.VISIBLE); 
357                      } 
358   
359                      ratingType.setVisibility(View.VISIBLE); 
360                      switch ((int) rating) { 
361                          case 1: 
362                              ratingType.setText(R.string.one_star); 
363                              break; 
364                          case 2: 
365                              ratingType.setText(R.string.two_star); 
366                              break; 
367                          case 3: 
368                              ratingType.setText(R.string.three_star);

374                              ratingType.setText(R.string.five_star); 
375                              break; 
376                          default: 
377                              break; 
378                      } 
379                  } 
380   
381              } 
382          });

384      }

386      @SuppressLint("ClickableViewAccessibility") 
387      private void ratingReviewsTouchListener() { 
388   
389          gestureDetector = new GestureDetectorCompat(getContext(), new GestureDetector.SimpleOnGestureListener() { 
390   
391              @Override 
392              public boolean onDown(MotionEvent e) { 
393                  return true; 
394              } 
395   
396              @Override 
397              public boolean onFling(MotionEvent e1, MotionEvent e2, float velocityX, float velocityY) { 
398                  boolean result = false; 
399                  try { 
400                      float diffY = e2.getY() - e1.getY(); 
401                      float diffX = e2.getX() - e1.getX(); 
402                      if (Math.abs(diffX) > Math.abs(diffY)) { 
403                          if (Math.abs(diffX) > SWIPE_THRESHOLD && Math.abs(velocityX) > SWIPE_VELOCITY_THRESHOLD) { 
404                              if (diffX > 0) { 
405                                  rightSwipe(); 
406                              } else { 
407                                  leftSwipe(); 
408                              } 
409                          } 
410                          result = true; 
411                      } else if (Math.abs(diffY) > SWIPE_THRESHOLD && Math.abs(velocityY) > SWIPE_VELOCITY_THRESHOLD) { 
412                          if (diffY > 0) { 
413  //                            Toast.makeText(getContext(), "Down", Toast.LENGTH_SHORT).show(); 
414                              viewPager.requestDisallowInterceptTouchEvent(false); 
415                              nestedScrollView.requestDisallowInterceptTouchEvent(false); 
416                          } else { 
417  //                            Toast.makeText(getContext(), "Up", Toast.LENGTH_SHORT).show(); 
418                              viewPager.requestDisallowInterceptTouchEvent(false); 
419                              nestedScrollView.requestDisallowInterceptTouchEvent(false); 
420                          } 
421                      } 
422                      result = true; 
423   
424                  } catch (Exception exception) { 
425                      exception.printStackTrace(); 
426                  } 
427                  return result; 
428              } 
429   
430          }); 
431   
432          swipeLayout.setOnTouchListener(new View.OnTouchListener() { 
433              @Override 
434              public boolean onTouch(View v, MotionEvent event) { 
435                  switch (event.getAction()) { 
436                      case MotionEvent.ACTION_DOWN: 
437                          viewPager.requestDisallowInterceptTouchEvent(true); 
438                          nestedScrollView.requestDisallowInterceptTouchEvent(true); 
439                          break; 
440                      case MotionEvent.ACTION_UP: 
441                          viewPager.requestDisallowInterceptTouchEvent(false); 
442                          nestedScrollView.requestDisallowInterceptTouchEvent(false); 
443                          break; 
444                  } 
445                  gestureDetector.onTouchEvent(event); 
446                  return true; 
447              } 
448   
449          }); 
450   
451      }

467          } else if (visibility(teachersRating)) { 
468              teachersRating.setVisibility(View.INVISIBLE); 
469              infrastructureRating.setVisibility(View.VISIBLE);

471          } else if (visibility(managementRating)) { 
472              managementRating.setVisibility(View.INVISIBLE); 
473              teachersRating.setVisibility(View.VISIBLE); 
474              selectedIndicator(techPath, mgmtPath); 
475          } else if (visibility(reviews)) { 
476              reviews.setVisibility(View.INVISIBLE); 
477              managementRating.setVisibility(View.VISIBLE);

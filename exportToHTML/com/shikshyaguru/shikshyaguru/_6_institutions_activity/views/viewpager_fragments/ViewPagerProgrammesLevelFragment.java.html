52       @Override 
53       public void setUpProgrammesLevel(InstitutionProgrammesData programmesData) { 
54           this.programmesData = programmesData; 
55           RecyclerView programmesLevelRecyclerView = rootView.findViewById(R.id.rec_inst_loader_vp_programmes_level); 
56           ProgrammesLevelAdapter adapter = new ProgrammesLevelAdapter(); 
57           programmesLevelRecyclerView.setLayoutManager(new LinearLayoutManager(getContext())); 
58           programmesLevelRecyclerView.setAdapter(adapter);

60       }

62       @Override 
63       public void onCoursesClickListener(String levelName, String courseName) { 
64           Intent intent = new Intent(getContext(), InstitutionsHomePageActivity.class); 
65           intent.putExtra("REQUEST_CODE", "courses_loader"); 
66           intent.putExtra("LEVEL_NAME", levelName); 
67           intent.putExtra("FACULTY_NAME", courseName); 
68           startActivity(intent); 
69       }

71       private class ProgrammesLevelAdapter extends RecyclerView.Adapter<ProgrammesLevelAdapter.ProgrammesLevelViewHolder> { 
72    
73           @NonNull 
74           @Override 
75           public ProgrammesLevelViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
76               View view = inflater.inflate(R.layout._6_2_2_1_rec_vp_programmes_level_item, parent, false); 
77               return new ProgrammesLevelViewHolder(view); 
78           } 
79    
80           @Override 
81           public void onBindViewHolder(@NonNull final ProgrammesLevelViewHolder holder, int position) { 
82               String programmesLabelName = programmesData.getProgrammesLevelNameList().get(position); 
83    
84               holder.programmesLevelName.setText(programmesLabelName.toUpperCase()); 
85               holder.recProgrammesLabel.setNestedScrollingEnabled(false); 
86               holder.recProgrammesLabel.setHasFixedSize(false); 
87    
88               final GridLayoutManager gridLayoutManager = new GridLayoutManager(getContext(), 2); 
89               holder.recProgrammesLabel.setLayoutManager(gridLayoutManager); 
90               ProgrammesFacultiesAdapter adapter = new ProgrammesFacultiesAdapter(); 
91   //            adapter.setFacultyName(programmesData.getProgrammesCoursesName().get(programmesLabelName)); 
92    
93               //Setting value in another adapter 
94               adapter.setFacultyName(programmesData.getProgrammesCoursesNameList().get(programmesLabelName)); 
95               adapter.setLevelName(programmesLabelName); 
96    
97               holder.recProgrammesLabel.setAdapter(adapter); 
98    
99               holder.layoutLevel.measure(View.MeasureSpec.makeMeasureSpec(holder.layoutLevel.getLayoutParams().width, View.MeasureSpec.EXACTLY), 
100                      View.MeasureSpec.makeMeasureSpec(0, View.MeasureSpec.UNSPECIFIED)); 
101              holder.anchorView.getLayoutParams().height = holder.layoutLevel.getMeasuredHeight(); 
102   
103          } 
104   
105          @Override 
106          public int getItemCount() { 
107  //            return programmesData.getProgrammesLevelName().length; 
108              return programmesData.getProgrammesLevelNameList().size(); 
109          }

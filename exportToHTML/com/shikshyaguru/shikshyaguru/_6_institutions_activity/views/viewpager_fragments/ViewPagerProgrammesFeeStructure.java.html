36       @Nullable 
37       @Override 
38       public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container, @Nullable Bundle savedInstanceState) { 
39           View view = inflater.inflate(R.layout._6_2_2_12_view_pager_programmes_fee_structure, container, false); 
40           this.inflater = inflater; 
41           this.rootView = view;

43           if (getArguments() != null) {

45               this.level = getArguments().getString("LEVEL_NAME"); 
46               this.faculty = getArguments().getString("FACULTY_NAME"); 
47               this.programme = getArguments().getString("COURSE_NAME"); 
48               this.opt_sem = getArguments().getString("OPT_SEM"); 
49               //noinspection unchecked 
50               this.fees = (List<HashMap<String, String>>) getArguments().getSerializable("FEE_LIST"); 
51    
52           } 
53    
54           String title = opt_sem.toUpperCase()+" - Fee structure "; 
55           Toolbar toolbar = view.findViewById(R.id.toolbar); 
56           Toolbars.setUpToolbar(toolbar, getActivity(), title); 
57           // To make onOptionItemSelected working we have to setHasOptionsMenu true in fragment. 
58           setHasOptionsMenu(true); 
59    
60    
61           setUpFeeStructure(); 
62    
63           return view; 
64       }

71       @Override 
72       public boolean onOptionsItemSelected(MenuItem item) { 
73    
74           switch (item.getItemId()) {

76               case android.R.id.home: 
77                   Objects.requireNonNull(getActivity()).onBackPressed();

83           }

86       public void setUpFeeStructure() { 
87           RecyclerView programmesLevelRecyclerView = rootView.findViewById(R.id.rec_inst_loader_vp_programmes_fee_structure); 
88           ProgrammesLevelAdapter adapter = new ProgrammesLevelAdapter(); 
89           programmesLevelRecyclerView.setLayoutManager(new LinearLayoutManager(getContext())); 
90           programmesLevelRecyclerView.setAdapter(adapter); 
91       } 
92    
93       private class ProgrammesLevelAdapter extends RecyclerView.Adapter<ProgrammesLevelAdapter.ProgrammesLevelViewHolder> { 
94    
95           @NonNull 
96           @Override 
97           public ProgrammesLevelViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
98               View view = inflater.inflate(R.layout._6_2_2_11_fee_structure_item, parent, false); 
99               return new ProgrammesLevelViewHolder(view); 
100          } 
101   
102          @Override 
103          public void onBindViewHolder(@NonNull final ProgrammesLevelViewHolder holder, int position) { 
104   
105              HashMap<String, String> currentItem = fees.get(position); 
106   
107              holder.feeName.setText(currentItem.get("type")); 
108              holder.amount.setText(currentItem.get("amount")); 
109   
110          } 
111   
112          @Override 
113          public int getItemCount() { 
114              return fees.size(); 
115          } 
116

118   
119              private TextView feeName, amount; 
120

122                  super(itemView); 
123   
124                  this.feeName = itemView.findViewById(R.id.lbl_fee_name); 
125                  this.amount = itemView.findViewById(R.id.lbl_amount); 
126   
127              } 
128   
129   
130          }

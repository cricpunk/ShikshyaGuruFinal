61       private void setUpAnchorView() { 
62    
63           RelativeLayout mainLayout = rootView.findViewById(R.id.rl_vp_gallery_main); 
64           View anchorView = rootView.findViewById(R.id.v_vp_gallery_anchor); 
65    
66           mainLayout.measure(View.MeasureSpec.makeMeasureSpec(mainLayout.getLayoutParams().width, View.MeasureSpec.EXACTLY), 
67                   View.MeasureSpec.makeMeasureSpec(0, View.MeasureSpec.UNSPECIFIED)); 
68    
69           anchorView.getLayoutParams().height = mainLayout.getMeasuredHeight(); 
70    
71           ViewGroup.MarginLayoutParams layoutParams = (ViewGroup.MarginLayoutParams) anchorView.getLayoutParams(); 
72           layoutParams.setMargins(100, 0, 100, 0); 
73           anchorView.setLayoutParams(layoutParams); 
74    
75       }

77       @Override 
78       public void setUpGalleryCategory(FirebaseRecyclerOptions<InstitutionGalleryData> options) { 
79    
80           RecyclerView galleryCategoryRecyclerView = rootView.findViewById(R.id.rec_inst_loader_vp_gallery); 
81           galleryCategoryRecyclerView.setNestedScrollingEnabled(false); 
82           RecyclerView.LayoutManager layoutManager = new LinearLayoutManager(getContext()); 
83           galleryCategoryRecyclerView.setLayoutManager(layoutManager); 
84           GalleryCategoryAdapter adapter = new GalleryCategoryAdapter(options); 
85           galleryCategoryRecyclerView.setAdapter(adapter); 
86    
87           adapter.startListening(); 
88       }

90       @Override 
91       public void onGalleryCategoryClick(String category, ArrayList<String> images, ArrayList<String> desc, ArrayList<String> ids) { 
92           Intent intent = new Intent(getContext(), InstitutionsHomePageActivity.class); 
93           intent.putExtra("REQUEST_CODE", "gallery_loader"); 
94           intent.putExtra("CATEGORY", category); 
95           intent.putStringArrayListExtra("IMAGES", images); 
96           intent.putStringArrayListExtra("DESCRIPTION", desc); 
97           intent.putStringArrayListExtra("IDS", ids); 
98           startActivity(intent); 
99    
100      }

102      private class GalleryCategoryAdapter extends FirebaseRecyclerAdapter<InstitutionGalleryData, GalleryCategoryAdapter.GalleryCategoryViewHolder> { 
103   
104          /* 
105           * Initialize a {@link RecyclerView.Adapter} that listens to a Firebase query. See 
106           * {@link FirebaseRecyclerOptions} for configuration options. 
107           * 
108           * @param options 
109           */ 
110          GalleryCategoryAdapter(@NonNull FirebaseRecyclerOptions<InstitutionGalleryData> options) { 
111              super(options); 
112          } 
113   
114          @Override 
115          protected void onBindViewHolder(@NonNull GalleryCategoryViewHolder holder, int position, @NonNull InstitutionGalleryData model) { 
116   
117              ViewGroup.MarginLayoutParams rootLP = (ViewGroup.MarginLayoutParams) holder.rootView.getLayoutParams(); 
118              ViewGroup.MarginLayoutParams categoryBgLP = (ViewGroup.MarginLayoutParams) holder.categoryBg.getLayoutParams(); 
119              ViewGroup.MarginLayoutParams imageBgLP = (ViewGroup.MarginLayoutParams) holder.imageBg.getLayoutParams(); 
120   
121              if (position % 2 == 0) { 
122                  rootLP.setMargins(0, 0, 150, 40); 
123                  categoryBgLP.setMargins(0,10,10,10); 
124                  imageBgLP.setMargins(116,0,0,0); 
125                  holder.rootView.setLayoutParams(rootLP); 
126                  holder.categoryBg.setLayoutParams(categoryBgLP); 
127                  holder.imageBg.setLayoutParams(imageBgLP); 
128              } else { 
129                  rootLP.setMargins(150, 0, 0, 40); 
130                  categoryBgLP.setMargins(10,10,0,10); 
131                  imageBgLP.setMargins(0,0,116,0); 
132                  holder.rootView.setLayoutParams(rootLP); 
133                  holder.categoryBg.setLayoutParams(categoryBgLP); 
134                  holder.imageBg.setLayoutParams(imageBgLP); 
135              } 
136   
137              // Category name of the current index 
138              String category = (new ArrayList<>(model.getCategoryWithImages().keySet())).get(position); 
139              // First image from the array to display in category view 
140              String image = model.getCategoryWithImages().get(category).get(0); 
141   
142              // Images, descriptions and ids of the current category 
143              // This will be transferred to gallery loader fragment trough intent 
144              holder.images = model.getCategoryWithImages().get(category); 
145              holder.desc = model.getCategoryWithImages().get(category); 
146              holder.ids = model.getCategoryWithImages().get(category); 
147              holder.category = category; 
148   
149              Picasso.get() 
150                      .load(image) 
151                      .fit() 
152                      .centerCrop() 
153                      .placeholder(R.drawable.logo_for_news) 
154                      .into(holder.categoryBg); 
155   
156              holder.categoryName.setText(category); 
157              holder.categoryTotalImage.setText(String.valueOf(model.getCategoryWithImages().get(category).size())); 
158   
159   
160          }

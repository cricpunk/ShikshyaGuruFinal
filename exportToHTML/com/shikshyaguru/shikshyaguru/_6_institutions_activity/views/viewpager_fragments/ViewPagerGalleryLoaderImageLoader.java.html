114      class GalleryPagerAdapter extends PagerAdapter { 
115   
116          @NonNull 
117          @Override 
118          public Object instantiateItem(@NonNull ViewGroup container, int position) { 
119   
120              View view = inflater.inflate(R.layout._6_2_5_5_full_size_image, container, false); 
121              SubsamplingScaleImageView imageView = view.findViewById(R.id.ivss_full_image); 
122   
123              loadImageByUrl(imageView, images.get(position), new okhttp3.OkHttpClient()); 
124   
125              container.addView(view); 
126              return view; 
127          } 
128   
129          void loadImageByUrl(SubsamplingScaleImageView scaleImageView, final String url, final okhttp3.OkHttpClient okHttpClient) { 
130   
131              scaleImageView.setMaxScale(5.0f); 
132              final Picasso picasso = Picasso.get(); 
133   
134              scaleImageView.setBitmapDecoderFactory(new DecoderFactory<ImageDecoder>() { 
135                  @NonNull 
136                  @Override 
137                  public ImageDecoder make() { 
138   
139                      return new PicassoDecoder(url, picasso); 
140                  } 
141              }); 
142   
143              scaleImageView.setRegionDecoderFactory(new DecoderFactory<ImageRegionDecoder>() { 
144                  @NonNull 
145                  @Override 
146                  public ImageRegionDecoder make() { 
147                      return new PicassoRegionDecoder(okHttpClient); 
148                  } 
149              }); 
150   
151              scaleImageView.setImage(ImageSource.uri(url)); 
152          } 
153   
154   
155          @Override 
156          public void destroyItem(@NonNull ViewGroup container, int position, @NonNull Object object) { 
157              container.removeView((RelativeLayout) object); 
158          } 
159   
160          @Override 
161          public int getCount() { 
162              return images.size(); 
163          } 
164   
165          @Override 
166          public boolean isViewFromObject(@NonNull View view, @NonNull Object object) { 
167              return view == object; 
168          }

228      private void hideDownloadIcon(final ArrowDownloadButton download) { 
229          new Handler().postDelayed(new Runnable() { 
230              @Override 
231              public void run() { 
232                  download.setVisibility(View.GONE); 
233              } 
234          }, 3500); 
235          resetDownloadIcon(download); 
236      }

239   
240          new Handler().postDelayed(new Runnable() {

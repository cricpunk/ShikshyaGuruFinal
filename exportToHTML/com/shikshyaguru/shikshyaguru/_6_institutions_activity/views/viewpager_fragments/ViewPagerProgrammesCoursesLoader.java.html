105      private class ProgrammesLevelAdapter extends RecyclerView.Adapter<ProgrammesLevelAdapter.ProgrammesLevelViewHolder> { 
106   
107          @NonNull 
108          @Override 
109          public ProgrammesLevelViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
110              View view = inflater.inflate(R.layout._6_2_2_8_rec_vp_programmes_courses_opener, parent, false); 
111              return new ProgrammesLevelViewHolder(view); 
112          } 
113   
114          @Override 
115          public void onBindViewHolder(@NonNull final ProgrammesLevelViewHolder holder, int position) { 
116   
117              String programmeName = programmesData.getOptionSemList().get(position); 
118   
119              holder.fees = programmesData.getFeeCollection().get(programmeName); 
120              holder.programmesLevelName.setText(programmeName.toUpperCase()); 
121   
122              holder.recProgrammesLabel.setNestedScrollingEnabled(false); 
123              holder.recProgrammesLabel.setHasFixedSize(false); 
124              holder.recProgrammesLabel.setLayoutManager(new LinearLayoutManager(getContext())); 
125   
126              ProgrammesFacultiesAdapter adapter = new ProgrammesFacultiesAdapter(); 
127   
128              //Setting value in another adapter 
129              adapter.setSubjects(programmesData.getSubjectCollection().get(programmeName)); 
130   
131              holder.recProgrammesLabel.setAdapter(adapter); 
132   
133              holder.layoutLevel.measure(View.MeasureSpec.makeMeasureSpec(holder.layoutLevel.getLayoutParams().width, View.MeasureSpec.EXACTLY), 
134                      View.MeasureSpec.makeMeasureSpec(0, View.MeasureSpec.UNSPECIFIED)); 
135              holder.anchorView.getLayoutParams().height = holder.layoutLevel.getMeasuredHeight(); 
136   
137          } 
138   
139          @Override 
140          public int getItemCount() { 
141              return programmesData.getOptionSemList().size(); 
142          }

144          class ProgrammesLevelViewHolder extends RecyclerView.ViewHolder { 
145   
146              private TextView programmesLevelName; 
147              private RecyclerView recProgrammesLabel; 
148              private LinearLayout layoutLevel; 
149              private View anchorView; 
150   
151              private List<HashMap<String, String>> fees; 
152   
153              ProgrammesLevelViewHolder(View itemView) { 
154                  super(itemView); 
155                  layoutLevel = itemView.findViewById(R.id.l_programmes_level); 
156                  anchorView = itemView.findViewById(R.id.v_inst_loader_vp_programmes_level); 
157                  programmesLevelName = itemView.findViewById(R.id.lbl_inst_loader_vp_programmes_label_name); 
158                  recProgrammesLabel = itemView.findViewById(R.id.rec_inst_loader_vp_programmes_courses); 
159   
160                  programmesLevelName.setOnClickListener(new View.OnClickListener() { 
161                      @Override 
162                      public void onClick(View v) { 
163                          onMoreIconClickListener(programmesLevelName); 
164                      } 
165                  }); 
166   
167              } 
168   
169              void onMoreIconClickListener(TextView textView) { 
170                  //Context contextWrapper = new android.view.ContextThemeWrapper(activity, R.style.teachersBusinessCardPopup); 
171                  PopupMenu popupMenu = new PopupMenu(Objects.requireNonNull(getContext()), textView, Gravity.END); 
172                  popupMenu.getMenuInflater().inflate(R.menu.courses_options_popup, popupMenu.getMenu()); 
173   
174                  popupMenu.setOnMenuItemClickListener(new PopupMenu.OnMenuItemClickListener() { 
175                      @Override 
176                      public boolean onMenuItemClick(MenuItem item) { 
177   
178                          switch (item.getItemId()) { 
179                              case R.id.courses_option_fee_structure: 
180                                  Intent intent = new Intent(getContext(), InstitutionsHomePageActivity.class); 
181                                  intent.putExtra("REQUEST_CODE", "fee_loader"); 
182                                  intent.putExtra("LEVEL_NAME", level); 
183                                  intent.putExtra("FACULTY_NAME", faculty); 
184                                  intent.putExtra("COURSE_NAME", programme); 
185                                  intent.putExtra("OPT_SEM", programmesLevelName.getText().toString()); 
186                                  intent.putExtra("FEE_LIST", (Serializable) fees); 
187                                  startActivity(intent); 
188                                  break; 
189                              case R.id.courses_option_routine: 
190                                  Toast.makeText(getContext(), "Will be updated soon..", Toast.LENGTH_SHORT).show(); 
191                                  break; 
192                              default: 
193                                  break; 
194                          } 
195   
196                          return true; 
197                      } 
198                  }); 
199                  popupMenu.show(); 
200              } 
201   
202          } 
203   
204      }

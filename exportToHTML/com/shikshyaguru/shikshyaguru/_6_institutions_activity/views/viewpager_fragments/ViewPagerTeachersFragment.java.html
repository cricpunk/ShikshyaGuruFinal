66       private class TeachersAdapter extends FirebaseRecyclerAdapter<InstitutionTeachersData, TeachersAdapter.TeachersViewHolder> { 
67    
68           /* 
69            * Initialize a {@link RecyclerView.Adapter} that listens to a Firebase query. See 
70            * {@link FirebaseRecyclerOptions} for configuration options. 
71            * 
72            * @param options 
73            */ 
74           TeachersAdapter(@NonNull FirebaseRecyclerOptions<InstitutionTeachersData> options) { 
75               super(options); 
76           } 
77    
78           @Override 
79           protected void onBindViewHolder(@NonNull TeachersViewHolder holder, int position, @NonNull InstitutionTeachersData model) { 
80    
81               holder.subjectContentIcon.setImageResource(R.drawable.ic_mind); 
82               holder.phoneContentIcon.setImageResource(R.drawable.ic_phone); 
83               holder.phoneContentIcon.setRotation(270); 
84               holder.emailContentIcon.setImageResource(R.drawable.ic_website); 
85    
86               Picasso.get() 
87                       .load(model.getImage_url()) 
88                       .fit() 
89                       .centerCrop() 
90                       .placeholder(R.drawable.ic_user) 
91                       .into(holder.image); 
92    
93               holder.name.setText(model.getFull_name()); 
94               holder.profession.setText(model.getDesignation()); 
95               holder.subjectContent1.setText(model.getExperiance()); 
96               //holder.subjectContent2.setText("Web Development"); 
97               holder.subjectContent2.setVisibility(View.GONE); 
98    
99               holder.phoneContent1.setText(model.getPhone_1()); 
100              if (!model.getPhone_2().equals("")) { 
101                  holder.phoneContent2.setText(model.getPhone_2()); 
102              } else { 
103                  holder.phoneContent2.setVisibility(View.GONE); 
104              } 
105   
106              holder.emailContent1.setText(model.getEmail_1()); 
107              if (!model.getEmail_2().equals("")) { 
108                  holder.emailContent2.setText(model.getEmail_2()); 
109              } else { 
110                  holder.emailContent2.setVisibility(View.GONE); 
111              } 
112   
113              holder.rating.setText(model.getRating()); 
114              holder.counter.setText(String.valueOf(position + 1)); 
115   
116          }

175              @Override 
176              public void onClick(View v) { 
177   
178                  Context wrapper = new android.view.ContextThemeWrapper(getActivity(), R.style.darkPopup); 
179                  PopupMenu popupMenu = new PopupMenu(wrapper, more, Gravity.END); 
180                  popupMenu.getMenuInflater().inflate(R.menu.teachers_business_card_popup, popupMenu.getMenu()); 
181   
182                  popupMenu.setOnMenuItemClickListener(new PopupMenu.OnMenuItemClickListener() { 
183                      @Override 
184                      public boolean onMenuItemClick(MenuItem item) { 
185                          switch (item.getItemId()) { 
186                              case R.id.teachers_card_popup_visit_profile: 
187                                  Toast.makeText(getContext(), "Visit profile", Toast.LENGTH_SHORT).show(); 
188                                  break; 
189                              case R.id.teachers_card_popup_send_message: 
190                                  Toast.makeText(getContext(), "Send message", Toast.LENGTH_SHORT).show(); 
191                                  break; 
192                              case R.id.teachers_card_popup_rate: 
193                                  Toast.makeText(getContext(), "Rate", Toast.LENGTH_SHORT).show(); 
194                                  break; 
195                              default: 
196                                  break; 
197   
198                          } 
199                          return true; 
200                      } 
201                  }); 
202                  popupMenu.show(); 
203   
204              }

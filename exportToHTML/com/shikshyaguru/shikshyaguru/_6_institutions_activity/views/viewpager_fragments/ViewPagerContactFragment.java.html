33   public class ViewPagerContactFragment extends Fragment implements 
34           View.OnFocusChangeListener, ViewPagerContactInterface { 
35    
36       private NestedScrollView currentLayout; 
37       private EditText name, phoneNumber, emailId, message; 
38       VPContactController controller; 
39    
40    
41       @Override 
42       public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container, @Nullable Bundle savedInstanceState) { 
43           return inflater.inflate(R.layout._6_2_10_0_view_pager_contact, container, false); 
44       } 
45    
46       @Override 
47       public void onViewCreated(@NonNull View view, @Nullable Bundle savedInstanceState) { 
48           super.onViewCreated(view, savedInstanceState); 
49           currentLayout = view.findViewById(R.id.current_layout); 
50           controller = new VPContactController(this, new InstitutionDataSource()); 
51           sendMessageSection(view); 
52       } 
53    
54    
55       private void sendMessageSection(View view) { 
56           name = view.findViewById(R.id.lbl_vp_contact_name); 
57           phoneNumber = view.findViewById(R.id.lbl_vp_contact_phone); 
58           emailId = view.findViewById(R.id.lbl_vp_contact_email); 
59           message = view.findViewById(R.id.lbl_vp_contact_message); 
60           FloatingActionButton sendMessageButton = view.findViewById(R.id.fab_vp_contact_send); 
61    
62           name.setOnFocusChangeListener(this); 
63           phoneNumber.setOnFocusChangeListener(this); 
64           emailId.setOnFocusChangeListener(this); 
65           message.setOnFocusChangeListener(this); 
66           sendMessageButton.setImageResource(R.drawable.ic_send); 
67    
68           sendMessageButton.setOnClickListener(new View.OnClickListener() { 
69               @Override 
70               public void onClick(View v) { 
71                   controller.onSendMessageButtonClick(); 
72               } 
73           }); 
74    
75       } 
76    
77    
78       @Override 
79       public void onFocusChange(View v, boolean hasFocus) { 
80           CircularReveal circularReveal = new CircularReveal(getActivity(), v, hasFocus); 
81           circularReveal.circularBackground(); 
82       } 
83    
84       @Override 
85       public void onSendMessageButtonClick() { 
86    
87           if (PerformValidation.inputFieldValidation(getActivity(), name) && 
88                   PerformValidation.inputFieldValidation(getActivity(), emailId) && 
89                   PerformValidation.inputFieldValidation(getActivity(), message) 
90                   ) { 
91    
92               if (InternetConnection.hasInternetConnection(Objects.requireNonNull(getContext()))) { 
93    
94                   String txtName = name.getText().toString(); 
95                   String txtPhone = phoneNumber.getText().toString(); 
96                   String txtEmail = emailId.getText().toString(); 
97                   String txtMessage = message.getText().toString(); 
98    
99                   InstitutionContactData contactData = new InstitutionContactData( 
100                          txtName, txtPhone, txtEmail, txtMessage 
101                  ); 
102   
103                  DatabaseError databaseError = controller.sendMessage(InstitutionLoaderFragment.id, contactData); 
104                  if ( databaseError == null ) { 
105                      PopupCollections.simpleSnackBar(currentLayout, "Message has been send successfully !", LoginFragment.COLOR_GREEN); 
106                      name.setText(""); 
107                      name.clearFocus(); 
108                      phoneNumber.setText(""); 
109                      emailId.setText(""); 
110                      message.setText(""); 
111                      message.clearFocus(); 
112                  } else { 
113                      PopupCollections.simpleSnackBar(currentLayout, databaseError.getMessage(), LoginFragment.COLOR_RED); 
114                  } 
115   
116              } else { 
117                  PopupCollections.simpleSnackBar(currentLayout, LoginFragment.NO_INTERNET_CONNECTION, LoginFragment.COLOR_GREEN); 
118              } 
119   
120          } 
121   
122      } 
123   
124  }

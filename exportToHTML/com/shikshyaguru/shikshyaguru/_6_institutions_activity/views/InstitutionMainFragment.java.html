97           instRecyclerView.setLayoutManager(new LinearLayoutManager(getContext(), LinearLayoutManager.VERTICAL, false));

105      @Override 
106      public void openInstitutionDetails(InstitutionsListItemParent instDetails, ActivityOptions options) { 
107   
108          String slogan = null; 
109   
110          Intent intent = new Intent(getContext(), InstitutionsHomePageActivity.class); 
111          intent.putExtra("REQUEST_CODE", "institutions_loader"); 
112          intent.putExtra("ID", instDetails.getId()); 
113          intent.putExtra("IMAGE", instDetails.getIcon_image()); 
114          intent.putExtra("NAME", instDetails.getName()); 
115          intent.putExtra("PLACE", instDetails.getCity()); 
116   
117          //noinspection ConstantConditions 
118          intent.putExtra("SLOGAN", slogan); 
119   
120          startActivity(intent, options.toBundle()); 
121      }

123      private class InstitutionAdapter extends FirebaseRecyclerAdapter<InstitutionsListItemParent, InstitutionAdapter.InstitutionViewHolder> { 
124   
125          /* 
126           * Initialize a {@link RecyclerView.Adapter} that listens to a Firebase query. See 
127           * {@link FirebaseRecyclerOptions} for configuration options. 
128           * 
129           * @param options 
130           */ 
131   
132          InstitutionAdapter(@NonNull FirebaseRecyclerOptions<InstitutionsListItemParent> options) { 
133              super(options); 
134          } 
135   
136          @Override 
137          protected void onBindViewHolder(@NonNull InstitutionAdapter.InstitutionViewHolder holder, int position, @NonNull InstitutionsListItemParent model) { 
138              //holder.institutionIcon.setImageResource(model.getIcon_image()); 
139              holder.institutionName.setText(model.getName()); 
140              holder.institutionRating.setText(model.getRating()); 
141              holder.institutionCityName.setText(model.getCity()); 
142              Picasso.get() 
143                      .load(model.getIcon_image()) 
144                      .placeholder(R.drawable.logo) 
145                      .into(holder.institutionIcon); 
146   
147              holder.instDetails = model; 
148   
149          } 
150   
151          @NonNull 
152          @Override 
153          public InstitutionAdapter.InstitutionViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
154              View view = inflater.inflate(R.layout._6_1_1_rec_inst_items, parent, false); 
155              return new InstitutionAdapter.InstitutionViewHolder(view); 
156          }

158          class InstitutionViewHolder extends RecyclerView.ViewHolder implements View.OnClickListener{ 
159   
160              private ViewGroup container; 
161              private ImageView institutionIcon; 
162              private TextView institutionName; 
163              private ImageView institutionMore; 
164              private TextView institutionRating; 
165              private TextView institutionCityName; 
166   
167              private InstitutionsListItemParent instDetails; 
168   
169              InstitutionViewHolder(View itemView) { 
170                  super(itemView); 
171   
172                  container = itemView.findViewById(R.id.root_institutions); 
173                  institutionIcon = itemView.findViewById(R.id.iv_institutions_icon); 
174                  institutionName = itemView.findViewById(R.id.lbl_institutions_name); 
175                  institutionMore = itemView.findViewById(R.id.iv_institutions_more); 
176                  institutionRating = itemView.findViewById(R.id.lbl_institutions_rating); 
177                  institutionCityName = itemView.findViewById(R.id.lbl_institutions_city_name); 
178   
179                  container.setOnClickListener(this); 
180                  institutionMore.setOnClickListener(this); 
181   
182              } 
183   
184              @Override 
185              public void onClick(View v) { 
186   
187                  switch (v.getId()) { 
188                      case R.id.root_institutions: 
189   
190                          Pair[] pairs = new Pair[2]; 
191                          pairs[0] = new Pair<View, String>(institutionIcon, "institutionImage"); 
192                          pairs[1] = new Pair<View, String>(institutionName, "institutionName"); 
193  //                        pairs[2] = new Pair<View, String>(institutionCityName, "institutionPlace"); 
194   
195                          //noinspection unchecked 
196                          ActivityOptions options = ActivityOptions.makeSceneTransitionAnimation(getActivity(), pairs); 
197   
198                          controller.onInstitutionItemClick(instDetails, options); 
199   
200                          break; 
201                      case R.id.iv_institutions_more: 
202   
203                          Toast.makeText(getContext(), "Clicked", Toast.LENGTH_SHORT).show(); 
204   
205                          break; 
206                      default:break; 
207                  } 
208   
209              } 
210          }

12   public class PrefManager { 
13    
14       private SharedPreferences pref; 
15       private SharedPreferences.Editor editor; 
16       //private Context _context; 
17    
18       // shared pref mode 
19       private static final int PRIVATE_MODE = 0; 
20       private static final String IS_FIRST_TIME_LAUNCH = "IsFirstTimeLaunch"; 
21       private static final String IS_USER_LOGGED_IN = "isUserLoggedIn"; 
22       private static final String IS_USER_TYPE_SET = "isUserTypeSet"; 
23    
24       public PrefManager(Context context, String preferenceName) { 
25           //this._context = context; 
26           pref = context.getSharedPreferences(preferenceName, PRIVATE_MODE); 
27           editor = pref.edit(); 
28           editor.apply(); 
29       } 
30    
31       public void setFirstTimeLaunch(boolean isFirstTime) { 
32           editor.putBoolean(IS_FIRST_TIME_LAUNCH, isFirstTime); 
33           editor.commit(); 
34       } 
35    
36       public boolean isFirstTimeLaunch() { 
37           return pref.getBoolean(IS_FIRST_TIME_LAUNCH, true); 
38       } 
39    
40       //Set user logged in status dynamically. This method will store users status into shared 
41       //preference 
42       public void setUserLoggedIn(boolean userLoggedIn) { 
43           editor.putBoolean(IS_USER_LOGGED_IN, userLoggedIn); 
44           editor.commit(); 
45       } 
46    
47       //By default set user logged in status true this method will used to check either user is 
48       // logged in or not. 
49       public boolean isUserLoggedIn() { 
50           return pref.getBoolean(IS_USER_LOGGED_IN, false); 
51       } 
52    
53       // User type is set for the single time. So set userType true when done for the first time 
54       public void setUserType(boolean userTypeSet) { 
55           editor.putBoolean(IS_USER_TYPE_SET, userTypeSet); 
56           editor.commit(); 
57       } 
58    
59       // Check either user type is set or not 
60       public boolean isUserTypeSet() { 
61           return pref.getBoolean(IS_USER_TYPE_SET, false); 
62       } 
63    
64   }

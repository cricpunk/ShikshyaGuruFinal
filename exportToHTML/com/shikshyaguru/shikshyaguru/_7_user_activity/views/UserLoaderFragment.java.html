138      private void initSlidingPanel() { 
139   
140          mainLayout.addPanelSlideListener(new SlidingUpPanelLayout.PanelSlideListener() { 
141              @Override 
142              public void onPanelSlide(View panel, float slideOffset) { 
143                  Log.i("Sliding Up", "onPanelSlide, offset " + slideOffset); 
144                  if (mainLayout != null) { 
145                      if (mainLayout.getAnchorPoint() == 1.0f) { 
146                          mainLayout.setAnchorPoint(0.19f); 
147                          mainLayout.setPanelState(SlidingUpPanelLayout.PanelState.ANCHORED); 
148                      } 
149                  } 
150   
151              } 
152   
153              @Override 
154              public void onPanelStateChanged(View panel, SlidingUpPanelLayout.PanelState previousState, SlidingUpPanelLayout.PanelState newState) { 
155                  Log.i("Sliding Up", "onPanelStateChanged " + newState); 
156              } 
157          });

168      @Override 
169      public void setUserProfile(Object userProfileDetails) { 
170   
171          Picasso.get() 
172                  .load(image) 
173                  .fit() 
174                  .centerCrop() 
175                  .placeholder(R.drawable.ic_user) 
176                  .into(fUserProfilePic); 
177   
178          fName.setText(name); 
179          fMail.setText(email); 
180   
181          if (type != null) { 
182              switch (type) { 
183                  case "1": 
184                      fType.setText(R.string.typeStudent); 
185                      break; 
186                  case "2": 
187                      fType.setText(R.string.typeTeacher); 
188                      break; 
189                  case "3": 
190                      fType.setText(R.string.typeInstitution); 
191                      break; 
192                  default: 
193                      break; 
194   
195              } 
196          } 
197          fFollower.setText(followers); 
198          fFollowing.setText(following); 
199          fInstitution.setText(institution); 
200   
201      }

203      @Override 
204      public void onClick(View v) { 
205   
206          switch (v.getId()) { 
207   
208              case R.id.btn_user_loader_send_message: 
209   
210                  if (fBtnSendMessage.getText().toString().toLowerCase().equals("update profile")) { 
211                      Toast.makeText(getContext(), "Update profile", Toast.LENGTH_SHORT).show(); 
212                  } else { 
213                      Toast.makeText(getContext(), "Send Message", Toast.LENGTH_SHORT).show(); 
214                  } 
215                  break; 
216   
217              case R.id.btn_user_loader_ask_question: 
218   
219                  if (fBtnAskQuestion.getText().toString().toLowerCase().equals("peoples")) { 
220   
221                      Toast.makeText(getContext(), "Peoples", Toast.LENGTH_SHORT).show(); 
222                      Intent intent = new Intent(getContext(), UserHomePageActivity.class); 
223                      intent.putExtra("REQUEST_CODE", "user_main"); 
224                      intent.putExtra("TITLE", "Users"); 
225                      intent.putExtra("CATEGORY", "all"); 
226                      startActivity(intent); 
227   
228                  } else { 
229                      Toast.makeText(getContext(), "Ask Question", Toast.LENGTH_SHORT).show(); 
230                  } 
231                  break; 
232   
233   
234          } 
235      }

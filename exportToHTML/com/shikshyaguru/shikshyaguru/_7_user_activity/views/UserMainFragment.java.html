184      class UserAdapter extends FirebaseRecyclerAdapter<UserDetails, UserViewHolder> { 
185   
186          /* 
187           * Initialize a {@link RecyclerView.Adapter} that listens to a Firebase query. See 
188           * {@link FirebaseRecyclerOptions} for configuration options. 
189           * 
190           * @param options 
191           */ 
192          UserAdapter(@NonNull FirebaseRecyclerOptions<UserDetails> options) { 
193              super(options); 
194          } 
195   
196          @Override 
197          protected void onBindViewHolder(@NonNull UserViewHolder holder, int position, @NonNull UserDetails model) { 
198   
199              try { 
200   
201                  Picasso.get() 
202                          .load(model.getImageUrl()) 
203                          .fit() 
204                          .centerCrop() 
205                          .placeholder(R.drawable.ic_user) 
206                          .into(holder.userIcon); 
207                  holder.userName.setText(model.getName()); 
208   
209                  if (model.getUserType() != null) { 
210                      holder.userType.setText(getUserType(model.getUserType())); 
211                  } 
212   
213                  holder.userUserName.setText(String.format("@%s", model.getUserName())); 
214                  holder.userInstitution.setText(model.getInstitution()); 
215                  changeButtonBehaviourAllUser(model.getuId(), holder.followBtn); 
216   
217                  holder.userDetails = model; 
218   
219              } catch (Exception e) { 
220                  showSnackbar(e.getMessage()); 
221              } 
222   
223          }

229              return new UserViewHolder(view);

311          @Override 
312          public void onClick(View v) { 
313   
314              switch (v.getId()) { 
315   
316                  case R.id.root_user: 
317   
318                      controller.openUserLoaderPage(userDetails); 
319   
320  //                    Intent intent = new Intent(getContext(), UserHomePageActivity.class); 
321  //                    intent.putExtra("REQUEST_CODE", "chat_loader"); 
322  //                    intent.putExtra("TITLE", userDetails.getName()); 
323  //                    intent.putExtra("UID", userDetails.getuId()); 
324  //                    startActivity(intent); 
325                      break; 
326   
327                  case R.id.btn_follow_following: 
328   
329                      switch (followBtn.getText().toString().toLowerCase()) { 
330                          case "follow": 
331                              controller.sendFollowRequest(userDetails.getuId()); 
332                              break; 
333                          case "following": 
334                              onFollowingBtnClick(userDetails.getuId(), userDetails.getName()); 
335                              break; 
336                          case "accept": 
337                              controller.acceptFollowingRequest(userDetails.getuId()); 
338                              break; 
339                          case "requested": 
340                              onRequestedBtnClick(userDetails.getuId(), userDetails.getName()); 
341                              break; 
342                          case "suggest": 
343                              controller.suggestInstitutionToFriend(userDetails.getuId(), institutionId); 
344                              break; 
345                      } 
346   
347                      break; 
348   
349                  default: 
350                      break; 
351   
352              } 
353   
354          }

439      private void changeButtonBehaviourAllUser(String uId, CircularProgressButton button) { 
440   
441          HashMap<String, Boolean> following = NavigationDrawerFragment.followingList; 
442   
443          if (following.containsKey(uId) && following.get(uId)) { 
444              button.setBackground(ContextCompat.getDrawable(Objects.requireNonNull(getContext()), R.drawable.rounded_bg)); 
445              button.setText(R.string.following); 
446              button.setTextColor(Color.parseColor("#E9E9E9")); 
447          } else if (following.containsKey(uId) && !following.get(uId)) { 
448              button.setText(R.string.requested); 
449          } 
450   
451      }

453      private void changeButtonBehaviour(HashMap<String, Boolean> userList, String uId, CircularProgressButton button) { 
454   
455          // If user is there in list key and value is true then set button as following 
456          // True means user is followed by current user 
457          if (userList.get(uId) != null && userList.get(uId)) { 
458   
459              switch (category) { 
460   
461                  case "followers": 
462                      if (NavigationDrawerFragment.followingList.containsKey(uId) && NavigationDrawerFragment.followingList.get(uId)) { 
463                          button.setBackground(ContextCompat.getDrawable(Objects.requireNonNull(getContext()), R.drawable.rounded_bg)); 
464                          button.setText(R.string.following); 
465                          button.setTextColor(Color.parseColor("#E9E9E9")); 
466                      } 
467                      break; 
468   
469                  case "following": 
470                      button.setBackground(ContextCompat.getDrawable(Objects.requireNonNull(getContext()), R.drawable.rounded_bg)); 
471                      button.setText(R.string.following); 
472                      button.setTextColor(Color.parseColor("#E9E9E9")); 
473                      break; 
474   
475                  case "suggest": 
476                      button.setText(R.string.suggest); 
477                      break; 
478   
479              } 
480   
481              // if user value in list is false then set button as requested. 
482              // Once request accepted button will be converted to following 
483          } else if (userList.get(uId) != null && !userList.get(uId)) { 
484   
485              switch (category) { 
486   
487                  case "followers": 
488                      button.setBackground(ContextCompat.getDrawable(Objects.requireNonNull(getContext()), R.drawable.rounded_bg)); 
489                      button.setText(R.string.accept); 
490                      button.setTextColor(Color.parseColor("#E9E9E9")); 
491                      break; 
492   
493                  case "following": 
494                      button.setText(R.string.requested); 
495                      break; 
496   
497              } 
498   
499   
500          }

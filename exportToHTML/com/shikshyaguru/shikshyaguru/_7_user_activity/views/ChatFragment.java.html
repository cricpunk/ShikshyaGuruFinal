140      private void sendMessageToFirebase(String senderId, String receiverId, String message) { 
141          chatList.clear(); 
142   
143          ChatMessage newMsg = new ChatMessage(senderId, receiverId, message); 
144          mMessagesDBRef.push().setValue(newMsg).addOnCompleteListener(new OnCompleteListener<Void>() { 
145              @Override 
146              public void onComplete(@NonNull Task<Void> task) { 
147                  if (!task.isSuccessful()) { 
148                      //error 
149                      Toast.makeText(getContext(), "Error " + task.getException().getLocalizedMessage(), Toast.LENGTH_SHORT).show(); 
150                  } else { 
151                      Toast.makeText(getContext(), "Message sent successfully!", Toast.LENGTH_SHORT).show(); 
152                      messageEditText.setText(null); 
153                      hideSoftKeyboard(); 
154                  } 
155              } 
156          }); 
157   
158      }

160      public void hideSoftKeyboard() { 
161          if (Objects.requireNonNull(getActivity()).getCurrentFocus() != null) { 
162              InputMethodManager inputMethodManager = (InputMethodManager) Objects.requireNonNull(getActivity()).getSystemService(INPUT_METHOD_SERVICE); 
163              assert inputMethodManager != null; 
164              inputMethodManager.hideSoftInputFromWindow(Objects.requireNonNull(Objects.requireNonNull(getActivity()).getCurrentFocus()).getWindowToken(), 0); 
165          } 
166      }

183      class ChatAdapter extends RecyclerView.Adapter<ChatViewHolder> { 
184   
185          public void add(int position, ChatMessage message) { 
186              chatList.add(position, message); 
187              notifyItemInserted(position); 
188          } 
189   
190          public void remove(int position) { 
191              chatList.remove(position); 
192              notifyItemRemoved(position); 
193          } 
194   
195          @NonNull 
196          @Override 
197          public ChatViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) { 
198   
199              View view = null; 
200   
201              if (viewType == ITEM_TYPE_SEND) { 
202                  view = inflater.inflate(R.layout._7_2_8_rec_send_message_bubble, parent, false); 
203              } else if (viewType == ITEM_TYPE_RECEIVED) { 
204                  view = inflater.inflate(R.layout._7_2_9_rec_received_message_bubble, parent, false); 
205              } 
206   
207              return new ChatViewHolder(view); 
208          } 
209   
210          @Override 
211          public void onBindViewHolder(@NonNull ChatViewHolder holder, int position) { 
212              ChatMessage chatMessage = chatList.get(position); 
213              holder.chatMessage.setText(chatMessage.getMessage()); 
214          } 
215   
216          @Override 
217          public int getItemCount() { 
218              return chatList.size(); 
219          } 
220   
221          @Override 
222          public int getItemViewType(int position) { 
223              if (chatList.get(position).getSender().equals(NavigationDrawerFragment.currentUser.getUid())) { 
224                  return ITEM_TYPE_SEND; 
225              } else { 
226                  return ITEM_TYPE_RECEIVED; 
227              } 
228          }

231      }

45       public FirebaseRecyclerOptions<UserDetails> getAllUsers(final UserMainInterface mainInterface, String category) { 
46    
47           Query query = mDatabase.getReference().child("users").orderByChild("profile"); 
48    
49           SnapshotParser<UserDetails> snapshotParser = new SnapshotParser<UserDetails>() { 
50               @NonNull 
51               @Override 
52               public UserDetails parseSnapshot(@NonNull DataSnapshot snapshot) { 
53    
54                   UserDetails userDetails = new UserDetails(); 
55    
56                   try { 
57    
58                       Long userType = snapshot.child("profile").child("type").getValue(Long.class); 
59    
60                       userDetails.setuId(snapshot.getKey()); 
61                       userDetails.setName(snapshot.child("profile").child("name").getValue(String.class)); 
62                       assert userType != null; 
63                       userDetails.setUserType(String.valueOf(userType.intValue())); 
64                       userDetails.setUserName(snapshot.child("profile").child("user_name").getValue(String.class)); 
65                       userDetails.setInstitution(snapshot.child("profile").child("institution").getValue(String.class)); 
66    
67                   } catch (Exception e) { 
68                       mainInterface.showSnackbar(e.getMessage()); 
69                   } 
70    
71                   mainInterface.removeSpinner(); 
72                   return userDetails; 
73    
74               } 
75           }; 
76    
77           return new FirebaseRecyclerOptions.Builder<UserDetails>().setQuery(query, snapshotParser).build(); 
78    
79       }

134      public void makeInstitutionSuggestionToFriend(final UserMainInterface mainInterface, final String friendId, final String institutionId) { 
135   
136          Query query = mDatabase.getReference().child("users").child(friendId).child("suggestions").orderByKey(); 
137          query.addListenerForSingleValueEvent(new ValueEventListener() { 
138              @Override 
139              public void onDataChange(DataSnapshot dataSnapshot) { 
140   
141                  // True means no previous suggestions 
142                  // False means suggestion has been made already 
143                  boolean validationPass = true; 
144   
145                  // Validate either same user has already suggested same institution or not 
146                  for (DataSnapshot snapshot : dataSnapshot.getChildren()) { 
147   
148                      String uIdRef = snapshot.child(uId).getKey(); 
149                      String instIdRef = snapshot.child(uId).getValue(String.class); 
150   
151                      if (uIdRef != null && Objects.equals(instIdRef, institutionId)) { 
152                          validationPass = false; 
153                          break; 
154                      } 
155   
156                  } 
157   
158                  if (validationPass) { 
159   
160                      HashMap<String, String> suggestion = new HashMap<>(); 
161                      suggestion.put(uId, institutionId); 
162                      mDatabase.getReference().child("users").child(friendId).child("suggestions").child(currentTime).setValue(suggestion, new DatabaseReference.CompletionListener() { 
163                          @Override 
164                          public void onComplete(DatabaseError databaseError, DatabaseReference databaseReference) { 
165                              if (databaseError == null) { 
166                                  mainInterface.showSnackbar("Suggestion has been made successfully !"); 
167                              } else { 
168                                  mainInterface.showSnackbar(databaseError.getMessage()); 
169                              } 
170                          } 
171                      }); 
172   
173                  } else { 
174                      mainInterface.showSnackbar("You have already suggested this institution !"); 
175                  } 
176   
177              } 
178   
179              @Override 
180              public void onCancelled(DatabaseError databaseError) { 
181                  mainInterface.showSnackbar(databaseError.getMessage()); 
182              } 
183          }); 
184   
185   
186      }

328      public void sendFollowRequest(final UserMainInterface mainInterface, String friendId) { 
329   
330          // In friend database table record will be stored at followers node 
331          DatabaseReference friendDBRef = mDatabase.getReference().child("users").child(friendId).child("followers").child(uId).child("status"); 
332   
333          // In own database table record will be stored at following node 
334          DatabaseReference ownDBRef = mDatabase.getReference().child("users").child(uId).child("following").child(friendId).child("status"); 
335   
336          friendDBRef.setValue(false); 
337          ownDBRef.setValue(false, new DatabaseReference.CompletionListener() {
